func addOperators(num string, target int) []string {
	ops := []byte("+-*")
	expressions := make([]string, 0, 1)
	var backtracking func(idx int, expression []byte, result int, product int)
	backtracking = func(idx int, expression []byte, result int, product int) {
		// ç»“å°¾
		if idx == len(num) {
			// æœ‰æ•ˆæ€§åˆ¤å®š
			if result+product == target {
				expressions = append(expressions, string(expression))
			}
			return
		}
		// ç”Ÿæˆå½“å‰é¡¹ï¼Œå‰å¯¼ 0 åˆ¤å®š
		for end := idx + 1; end <= len(num) && (end-idx <= 1 || num[idx] != '0'); end++ {
			t, _ := strconv.ParseInt(num[idx:end], 10, 64)
			elem := int(t)
			// é¦–é¡¹åˆ¤å®š
			if len(expression) == 0 {
				expression = append(expression, num[idx:end]...)
				backtracking(end, expression, 0, elem)
				expression = expression[:0] // å›æº¯
				continue
			}
			// ä¸‰ç§è¿ç®—ç¬¦
			for _, op := range ops {
				expression = append(expression, op)
				expression = append(expression, num[idx:end]...)
				switch op {
				case '*':
					backtracking(end, expression, result, product*elem)
				case '-':
					backtracking(end, expression, result+product, -elem)
				case '+':
					backtracking(end, expression, result+product, elem)
				}
				expression = expression[:len(expression)-1-end+idx] // å›æº¯
			}
		}
	}
	backtracking(0, make([]byte, 0, len(num)<<2), 0, 0)
	return expressions
}
