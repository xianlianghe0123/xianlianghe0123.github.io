func getSum(a int, b int) int {
	c := 0
	ans := 0
	for bit := 1; bit != 0; bit <<= 1 {
		ans |= (a ^ b ^ c) & bit
		c = (((a ^ b) & c) | (a & b)) & bit << 1
	}
	return ans
}
