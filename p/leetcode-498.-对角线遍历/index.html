func findDiagonalOrder(mat [][]int) []int {
	rows, cols := len(mat), len(mat[0])
	// åˆ¤æ–­æ˜¯å¦åœ¨çŸ©é˜µå†…
	var valid = func(row int, col int) bool {
		if row < 0 || row >= rows || col < 0 || col >= cols {
			return false
		}
		return true
	}

	ans := make([]int, 0, rows*cols)
	rightUp := true
	for row, col := 0, 0; ; {
		// å‘ç”Ÿè¶Šç•Œ
		if !valid(row, col) {
			if rightUp {
				// å³ä¸Šæ–¹è¶Šç•Œï¼Œ
				row++
				if !valid(row, col) {
					row++
					col--
				}
			} else {
				// å·¦ä¸‹æ–¹è¶Šç•Œ
				col++
				if !valid(row, col) {
					row--
					col++
				}

			}
			rightUp = !rightUp // å˜æ¢æ–¹å‘
		}
		ans = append(ans, mat[row][col])
		// éå†ç»“æŸ
		if row == rows-1 && col == cols-1 {
			break
		}
		if rightUp {
			row--
			col++
		} else {
			row++
			col--
		}
	}

	return ans
}
