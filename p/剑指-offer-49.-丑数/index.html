type MinHeap struct {
	sort.IntSlice
}

func (this *MinHeap) Push(x interface{}) {
	this.IntSlice = append(this.IntSlice, x.(int))
}

func (this *MinHeap) Pop() interface{} {
	x := this.IntSlice[this.Len()-1]
	this.IntSlice = this.IntSlice[:this.Len()-1]
	return x
}

func nthUglyNumber(n int) int {
    exist := make(map[int]struct{})
    factors := []int{2,3,5}
	h := &MinHeap{
		IntSlice: make([]int, 0, n*3),
	}
	heap.Push(h, 1)
	ans := 0
	for i := 0; i < n; i++ {
		ans = heap.Pop(h).(int)
        for _, factor := range factors {
            if _, ok := exist[ans*factor]; !ok {
		        heap.Push(h, ans*factor)
                exist[ans*factor] = struct{}{}
            }
        }
	}
	return ans
}

func nthUglyNumber(n int) int {
	dp := make([]int, n)
	dp[0] = 1
	for i := 0; i < n; i++ {
		t2, t3, t5 = dp[p2]*2, dp[p3]*3, dp[p5]*5
		dp[i] = t2
		if dp[i] > t3 {
			dp[i] = t3
		}
		if dp[i] > t5 {
			dp[i] = t5
		}
		if dp[i] == t2 {
			p2++
		}
		if dp[i] == t3 {
			p3++
		}
		if dp[i] == t5 {
			p5++
		}
	}
	return dp[n-1]
}
