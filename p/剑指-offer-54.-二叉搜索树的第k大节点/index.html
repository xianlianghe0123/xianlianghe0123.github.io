func kthLargest(root *TreeNode, k int) int {
	cnt := 0
	ans := 0
	var dfs func(node *TreeNode) bool
	dfs = func(node *TreeNode) bool {
		if node == nil {
			return false
		}
		if dfs(node.Right) {
			return true
		}
		cnt++
		if cnt == k {
			ans = node.Val
			return true
		}
		return dfs(node.Left)
	}
	dfs(root)
	return ans
}
