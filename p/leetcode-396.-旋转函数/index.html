func maxRotateFunction(nums []int) int {
	sum := 0
	f := 0
	for i, num := range nums {
		sum += num
		f += i * num
	}
	maxF := f
	for i := len(nums) - 1; i > 0; i-- {
		f += sum - nums[i]*len(nums)
		if maxF < f {
			maxF = f
		}
	}
	return maxF
}
