func maxRepeating(sequence string, word string) int {
	ans := 0
	dp := make([]int, len(sequence)+1)
	for i := len(word); i <= len(sequence); i++ {
		if strings.HasSuffix(sequence[:i], word) {
			dp[i] = dp[i-len(word)] + 1
			if ans < dp[i] {
				ans = dp[i]
			}
		}
	}
	return ans
}
