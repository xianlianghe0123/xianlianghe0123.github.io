func levelOrder(root *TreeNode) [][]int {
	ans := make([][]int, 0)
	if root == nil {
		return ans
	}
	q := list.New()
	q.PushBack(root)
	for q.Len() > 0 {
        // å…ˆå¶æ•°å±‚
		tmp := make([]int, 0, 1)
		for i := q.Len(); i > 0 ; i-- {
			node := q.Front().Value.(*TreeNode)
			q.Remove(q.Front())
			tmp = append(tmp, node.Val)
			if node.Left != nil {
				q.PushBack(node.Left)
			}
			if node.Right != nil {
				q.PushBack(node.Right)
			}
		}
        ans = append(ans, tmp)
        // ç©ºè·³è¿‡
        if q.Len() == 0 {
            break
        }
        // åå¥‡æ•°å±‚
		tmp = make([]int, 0, 1)
		for i := q.Len(); i > 0 ; i-- {
			node := q.Back().Value.(*TreeNode)
			q.Remove(q.Back())
			tmp = append(tmp, node.Val)
			if node.Right != nil {
				q.PushFront(node.Right)
			}
			if node.Left != nil {
				q.PushFront(node.Left)
			}
		}
        ans = append(ans, tmp)
	}
	return ans
}
