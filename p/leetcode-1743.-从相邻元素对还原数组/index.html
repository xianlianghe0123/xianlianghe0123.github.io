func restoreArray(adjacentPairs [][]int) []int {
	to := make(map[int][]int)
	for _, pair := range adjacentPairs {
		to[pair[0]] = append(to[pair[0]], pair[1])
		to[pair[1]] = append(to[pair[1]], pair[0])
	}
	res := make([]int, len(adjacentPairs)+1)
	for k, v := range to {
		if len(v) == 1 {
			res[0] = k
			res[1] = v[0]
			break
		}
	}
	for i := 2; i < len(res); i++ {
		next := to[res[i-1]]
		if next[0] != res[i-2] {
			res[i] = next[0]
		} else {
			res[i] = next[1]
		}
	}
	return res
}
