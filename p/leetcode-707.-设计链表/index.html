type node struct {
	val  int
	next *node
}

type MyLinkedList struct {
	head   *node
	length int
}

func Constructor() MyLinkedList {
	return MyLinkedList{
		head:   new(node),
		length: 0,
	}
}

func (this *MyLinkedList) Get(index int) int {
	if index < -1 || index >= this.length {
		return -1
	}
	n := this.head
	for ; index >= 0; index-- {
		n = n.next
	}
	return n.val
}

func (this *MyLinkedList) AddAtHead(val int) {
	this.head.next = &node{
		val:  val,
		next: this.head.next,
	}
	this.length++
}

func (this *MyLinkedList) AddAtTail(val int) {
	n := this.head
	for ; n.next != nil; n = n.next {
	}
	n.next = &node{
		val:  val,
		next: n.next,
	}
	this.length++
}

func (this *MyLinkedList) getPre(index int) *node {
	// ÂØªÊâæ‰∏ä‰∏Ä‰∏™ËäÇÁÇπ
	n := this.head
	for ; index > 0; index-- {
		n = n.next
	}
	return n
}

func (this *MyLinkedList) AddAtIndex(index int, val int) {
	if index < 0 {
		this.AddAtHead(val)
		return
	}
	if index > this.length {
		return
	}
	n := this.getPre(index)
	n.next = &node{
		val:  val,
		next: n.next,
	}
	this.length++
}

func (this *MyLinkedList) DeleteAtIndex(index int) {
	if index < 0 || index >= this.length {
		return
	}
	n := this.getPre(index)
	n.next = n.next.next
	this.length--
}
