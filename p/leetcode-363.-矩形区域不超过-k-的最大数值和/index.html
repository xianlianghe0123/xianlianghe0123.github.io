func compareVersion(version1 string, version2 string) int {
	num1 := strings.Split(version1, ".")

	for i := 0; i < len(num1) || i < len(num2); i++ {
		var p, q int64
		if i < len(num1) {
			p, _ = strconv.ParseInt(num1[i], 10, 64)
		}
		if i < len(num2) {
			q, _ = strconv.ParseInt(num2[i], 10, 64)
		}
		if p > q {
			return 1
		} else if p < q {
			return -1
		}
	}
	return 0
}
