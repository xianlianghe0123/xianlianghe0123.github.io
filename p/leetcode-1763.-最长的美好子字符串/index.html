func longestNiceSubstring(s string) string {
	result := ""
	for begin := 0; begin < len(s); begin++ {
		upper, lower := 0, 0
		for i := begin; i < len(s); i++ {
			if s[i] >= 'a' && s[i] <= 'z' {
				lower |= 1 << (s[i] - 'a')
			} else {
				upper |= 1 << (s[i] - 'A')
			}
			if lower == upper && len(result) < i-begin+1 {
				result = s[begin : i+1]
			}
		}
	}
	return result
}
