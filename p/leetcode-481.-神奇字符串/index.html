func magicalString(n int) int {
	if n <= 2 {
		return 1
	}
	buf := make([]byte, 0, n)
	buf = append(buf, 1, 2, 2)
	ans := 1
	for i, char := 2, byte(1); len(buf) < n; i++ {
		before := len(buf)
		buf = append(buf, char)
		if buf[i] == 2 && len(buf) < n {
			buf = append(buf, char)
		}
		if char == 1 {
			ans += len(buf) - before
		}
		char ^= 0b11
	}
	return ans
}
