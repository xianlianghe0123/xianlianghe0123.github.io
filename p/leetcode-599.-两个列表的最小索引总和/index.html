func findRestaurant(list1 []string, list2 []string) []string {
	sum := math.MaxInt
	ans := make([]string, 0)
	for i, l1 := range list1 {
		for j, l2 := range list2 {
			if l1 == l2 {
				if sum > i+j {
					sum = i + j
					ans = []string{l1}
				} else if sum == i+j {
					ans = append(ans, l1)
				}
			}
		}
	}
	return ans
}

func findRestaurant(list1 []string, list2 []string) []string {
	idx := make(map[string]int)
	for i, l := range list1 {
		idx[l] = i
	}
	sum := math.MaxInt
	ans := make([]string, 0)
	for i, l := range list2 {
		if _, ok := idx[l]; ok {
			if sum > idx[l]+i {
				sum = idx[l] + i
				ans = []string{l}
			} else if sum == idx[l]+i {
				ans = append(ans, l)
			}
		}
	}
	return ans
}
