func defangIPaddr(address string) string {
	ans := make([]byte, 0, len(address)+6)
	for i := range address {
		if address[i] != '.' {
			ans = append(ans, address[i])
			continue
		}
		ans = append(ans, '[', '.', ']')
	}
	return string(ans)
}
