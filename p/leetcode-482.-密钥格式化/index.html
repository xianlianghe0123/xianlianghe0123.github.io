func licenseKeyFormatting(s string, k int) string {
	ans := make([]byte, 0, len(s)<<1)
	for i, cnt := len(s)-1, 0; i >= 0; i-- {
		if s[i] == '-' {
			continue
		}
		t := s[i]
		if 'a' <= t && t <= 'z' {
			t = t - 'a' + 'A'
		}
		ans = append(ans, t)
		cnt++
		if cnt == k {
			ans = append(ans, '-')
			cnt = 0
		}
	}
	for i := len(ans)/2 - 1; i >= 0; i-- {
		ans[i], ans[len(ans)-1-i] = ans[len(ans)-1-i], ans[i]
	}
	if len(ans) > 0 && ans[0] == '-' {
		ans = ans[1:]
	}
	return string(ans)
}
