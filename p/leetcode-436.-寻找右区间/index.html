func findRightInterval(intervals [][]int) []int {
	idx := make(map[int]int)
	start := make([]int, 0, len(intervals))
	for i, interval := range intervals {
		idx[interval[0]] = i
		start = append(start, interval[0])
	}
	sort.Ints(start)
	ans := make([]int, 0, len(intervals))
	for _, interval := range intervals {
		i := sort.SearchInts(start, interval[1])
		// æ‰¾ä¸åˆ°å¤§äºç­‰äº interval[1]
		if i == len(start) {
			ans = append(ans, -1)
		} else {
			ans = append(ans, idx[start[i]])
		}
	}
	return ans
}
