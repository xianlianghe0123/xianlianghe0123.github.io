type MaxHeap struct {
	sort.IntSlice
}

func (this *MaxHeap) Less(i, j int) bool {
	return this.IntSlice.Less(j, i)
}

func (this *MaxHeap) Push(x interface{}) {
	this.IntSlice = append(this.IntSlice, x.(int))
}

func (this *MaxHeap) Pop() interface{} {
	x := this.IntSlice[0]
	this.IntSlice = this.IntSlice[:this.Len()-1]
	return x
}

func (this *MaxHeap) Top() int {
	return this.IntSlice[0]
}

func getLeastNumbers(arr []int, k int) []int {
	h := &MaxHeap{IntSlice: make([]int, 0, k)}
	for _, num := range arr {
		if h.Len() < k {
			heap.Push(h, num)
		} else if h.Len() > 0 && h.Top() > num {
			heap.Pop(h)
			heap.Push(h, num)
		}
	}
	return h.IntSlice
}

func getLeastNumbers(arr []int, k int) []int {
	pos := 0
	for i := range arr {
		if arr[i] <= arr[len(arr)-1] {
			arr[i], arr[pos] = arr[pos], arr[i]
			pos++
		}
	}
	if pos == k {
		return arr[:k]
	} else if pos < k {
		return append(arr[:pos], getLeastNumbers(arr[pos:], k-pos)...)
	}
	return getLeastNumbers(arr[:pos-1], k)
}
