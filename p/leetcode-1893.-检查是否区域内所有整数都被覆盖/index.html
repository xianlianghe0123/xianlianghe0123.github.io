func isCovered(ranges [][]int, left int, right int) bool {
	isCover := [51]bool{}
	for _, r := range ranges {
		for i := r[0]; i <= r[1]; i++ {
			isCover[i] = true
		}
	}
	for i := left; i <= right; i++ {
		if !isCover[i] {
			return false
		}
	}
	return true
}
