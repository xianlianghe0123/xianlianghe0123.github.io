func findRadius(houses []int, heaters []int) int {
	ans := 0
	for _, house := range houses {
		radius := math.MaxInt32
		for _, heater := range heaters {
			if heater > house {
				if radius > heater-house {
					radius = heater - house
				}
			} else {
				if radius > house-heater {
					radius = house - heater
				}
			}
		}
		if ans < radius {
			ans = radius
		}
	}
	return ans
}

func findRadius(houses []int, heaters []int) int {
	sort.Ints(heaters)
	ans := 0
	for _, house := range houses {
		radius := math.MaxInt32
		for l, r := 0, len(heaters)-1; l <= r; {
			mid := (l + r) / 2
			if heaters[mid] == house {
				radius = 0
				break
			} else if heaters[mid] < house {
				radius = min(radius, house-heaters[mid])
				l = mid + 1
			} else {
				radius = min(radius, heaters[mid]-house)
				r = mid - 1
			}
		}
		if ans < radius {
			ans = radius
		}
	}
	return ans
}
