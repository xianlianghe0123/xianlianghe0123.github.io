func findLongestWord(s string, dictionary []string) string {
	contains := func(s, sub string) bool {
		j := 0
		for i := 0; i < len(s) && j < len(sub); i++ {
			if s[i] == sub[j] {
				j++
			}
		}
		if j == len(sub) {
			return true
		}
		return false
	}
	maxLen := 0
	ans := ""
	for _, d := range dictionary {
		if contains(s, d) {
			if maxLen < len(d) {
				maxLen = len(d)
				ans = d
			} else if maxLen == len(d) && ans > d {
				ans = d
			}
		}
	}
	return ans
}
