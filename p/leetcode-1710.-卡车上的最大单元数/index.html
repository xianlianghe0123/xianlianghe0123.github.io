func maximumUnits(boxTypes [][]int, truckSize int) int {
	sort.Slice(boxTypes, func(i, j int) bool {
		return boxTypes[i][1] > boxTypes[j][1]
	})
	ans := 0
	for _, b := range boxTypes {
		if truckSize <= b[0] {
			ans += truckSize * b[1]
			break
		}
		truckSize -= b[0]
		ans += b[0] * b[1]
	}
	return ans
}
