func getHint(secret string, guess string) string {
	secretCnt := make(map[byte]int)
	guessCnt := make(map[byte]int)
	bulls := 0
	for i := range secret {
		if secret[i] == guess[i] {
			bulls++
		} else {
			secretCnt[secret[i]]++
			guessCnt[guess[i]]++
		}
	}
	cows := 0
	for char := range guessCnt {
		if guessCnt[char] < secretCnt[char] {
			cows += guessCnt[char]
		} else {
			cows += secretCnt[char]
		}
	}
	return fmt.Sprintf("%dA%dB", bulls, cows)
}
