func factorial(n int64, mod int64) int64 {
	ans := int64(1)
	for i := int64(2); i <= n; i++ {
		ans = (ans * i) % mod
	}
	return ans
}

func isPrime(x int) bool {
	for i := int(math.Sqrt(float64(x))); i > 1; i-- {
		if x%i == 0 {
			return true
		}
	}
	return false
}

func numPrimeArrangements(n int) int {
	const MOD = int64(1e9 + 7)
	// ÁªüËÆ°Ë¥®Êï∞Êï∞Èáè
	numPrimes := 0
	for i := 2; i <= n; i++ {
		if !isPrime(i) {
			numPrimes++
		}
	}
	return int(factorial(int64(numPrimes), MOD) * factorial(int64(n-numPrimes), MOD) % MOD)
}
