func findBall(grid [][]int) []int {
	ans := make([]int, len(grid[0]))
	for i := 0; i < len(grid[0]); i++ {
		col := i
		for row := 0; row < len(grid); row++ {
			if grid[row][col] == 1 {
				// Âç°‰Ωè
				if col+1 == len(grid[0]) || grid[row][col+1] == -1 {
					col = -1
					break
				}
				col++
			} else {
				// Âç°‰Ωè
				if col == 0 || grid[row][col-1] == 1 {
					col = -1
					break
				}
				col--
			}
		}
		ans[i] = col
	}
	return ans
}
