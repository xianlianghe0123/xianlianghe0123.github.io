func findClosest(words []string, word1 string, word2 string) int {
	min := math.MaxInt
	idx1 := -1
	idx2 := -1
	for i, word := range words {
		if word == word1 {
			idx1 = i
		}
		if word == word2 {
			idx2 = i
		}
		if idx1 == -1 || idx2 == -1 {
			continue
		}
		d := idx1 - idx2
		if d < 0 {
			d = -d
		}
		if min > d {
			min = d
		}
	}
	return min
}
