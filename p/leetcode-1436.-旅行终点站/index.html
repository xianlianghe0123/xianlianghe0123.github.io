func destCity(paths [][]string) string {
	outDegrees := make(map[string]int)
	for _, path := range paths {
		outDegrees[path[0]]++
		outDegrees[path[1]] += 0
	}
	for city, outDegree := range outDegrees {
		if outDegree == 0 {
			return city
		}
	}
	return ""
}
