func diameterOfBinaryTree(root *TreeNode) int {
	var dfs func(node *TreeNode) (int, int)
	dfs = func(node *TreeNode) (int, int) {
		if node == nil {
			return 0, 0
		}
		lDepth, lMax := dfs(node.Left)
		rDepth, rMax := dfs(node.Right)
        depth := lDepth
        if depth < rDepth {
            depth = rDepth
        }
        max := lDepth + rDepth
        if max < lMax {
            max = lMax
        }
        if max < rMax {
            max = rMax
        }
        return depth+1, max
	}
	_, max := dfs(root)
	return max
}
