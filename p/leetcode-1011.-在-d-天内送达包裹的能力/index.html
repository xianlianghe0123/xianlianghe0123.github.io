func shipWithinDays(weights []int, D int) int {
	r := 0
	l := 0
	for _, weight := range weights {
		r += weight
		if l < weight {
			l = weight
		}
	}

	var binarySearch func(l, r int) int
	binarySearch = func(l, r int) int {
		if l == r {
			return l
		}
		mid := (l + r) / 2
		cnt := 1
		curCap := mid
		for i := 0; i < len(weights); {
			// èƒ½è£…å…¥ï¼Œç»§ç»­
			if curCap >= weights[i] {
