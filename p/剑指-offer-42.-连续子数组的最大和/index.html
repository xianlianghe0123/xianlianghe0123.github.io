func maxSubArray(nums []int) int {
	if len(nums) == 0 {
		return 0
	}
	dp := nums[0]
    max := dp
	for pre, i := 0, 1; i < len(nums); i++ {
		pre = dp
		if pre+nums[i] > nums[i] {
			dp = pre + nums[i]
		} else {
			dp = nums[i]
		}
        if max < dp {
            max = dp
        }
	}
	return max
}
