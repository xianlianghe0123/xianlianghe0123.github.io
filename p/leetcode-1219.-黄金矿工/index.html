func getMaximumGold(grid [][]int) (maxGold int) {
	rows, cols := len(grid), len(grid[0])
	dirs := [][]int{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}
	var dfs func(row, col, gold int)
	dfs = func(row, col, gold int) {
		// Ë∂äÁïåÂà§Êñ≠
		if row < 0 || row >= rows || col < 0 || col >= cols {
			return
		}
		// ‰∏çÂèØËøõÂÖ•
		if grid[row][col] == 0 {
			return
		}
		dup := grid[row][col]
		gold += dup // ÂºÄÈáá
		// ËÆ∞ÂΩïÊúÄÂ§ßÂÄº
		if maxGold < gold {
			maxGold = gold
		}
		grid[row][col] = 0 // Ê†áËÆ∞Â∑≤ÂºÄÈáá
		// ÂØªÊâæÂêéÁª≠Ë∑ØÂæÑ
		for _, d := range dirs {
			dfs(row+d[0], col+d[1], gold)
		}
		grid[row][col] = dup // ÂõûÊ∫Ø

	}
	// ‰ª•ÊØè‰∏™ÂçïÂÖÉ‰∏∫Ëµ∑ÂßãÔºåÂõûÊ∫ØÊü•Êâæ
	for row, line := range grid {
		for col, cell := range line {
			if cell > 0 {
				dfs(row, col, 0)
			}
		}
	}
	return
}
