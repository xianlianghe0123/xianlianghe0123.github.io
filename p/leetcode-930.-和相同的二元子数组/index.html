func numSubarraysWithSum(nums []int, goal int) int {
	cnt := make(map[int]int)
	res := 0
	sum := 0
	for _, num := range nums {
		cnt[sum]++
		sum += num
		res += cnt[sum-goal]
	}
	return res
}
