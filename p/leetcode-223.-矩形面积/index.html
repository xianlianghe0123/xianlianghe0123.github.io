func computeArea(ax1 int, ay1 int, ax2 int, ay2 int, bx1 int, by1 int, bx2 int, by2 int) int {
	area := (ax2-ax1)*(ay2-ay1) + (bx2-bx1)*(by2-by1)
	x1 := ax1
	if x1 < bx1 {
		x1 = bx1
	}
	y1 := ay1
	if y1 < by1 {
		y1 = by1
	}
	x2 := ax2
	if x2 > bx2 {
		x2 = bx2
	}
	y2 := ay2
	if y2 > by2 {
		y2 = by2
	}
	if x1 <= x2 && y1 <= y2 {
		area -= (x2 - x1) * (y2 - y1)
	}
	return area
}
