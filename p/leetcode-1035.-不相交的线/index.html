    func maxUncrossedLines(nums1 []int, nums2 []int) int {
	dp := make([][]int, 0, len(nums1)+1)
	dp = append(dp, make([]int, len(nums2)+1))
	for i := range nums1 {
		dp = append(dp, make([]int, len(nums2)+1))
		for j := range nums2 {
			if nums1[i] == nums2[j] {
				dp[i+1][j+1] = dp[i][j] + 1
			} else {
				if dp[i][j+1] > dp[i+1][j] {
					dp[i+1][j+1] = dp[i][j+1]
				} else {
					dp[i+1][j+1] = dp[i+1][j]
				}
			}
		}
	}
	return dp[len(nums1)][len(nums2)]
}
