func reformatNumber(number string) string {
	ans := make([]byte, 0, len(number))
	cnt := 0
	for i := range number {
		if number[i] < '0' || number[i] > '9' {
			continue
		}
		ans = append(ans, number[i])
		cnt = (cnt + 1) % 3
		// ÊØè‰∏â‰∏™Êú´Â∞æÂä†‰∏Ä‰∏™ -
		if cnt == 0 {
			ans = append(ans, '-')
		}
	}
	switch cnt {
	case 0:
		ans = ans[:len(ans)-1]
	case 1:
		ans[len(ans)-3], ans[len(ans)-2] = ans[len(ans)-2], ans[len(ans)-3]
	}
	return string(ans)
}
