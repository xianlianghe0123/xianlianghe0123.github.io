func change(amount int, coins []int) int {
	dp = make([]int, amount+1)
	dp[0] = 1
	for i := range coins {
		for j := 0; j <= amount; j++ {
			for k := 0; k*coins[i] <= j; k++ {
				dp[j] += dp[j-k*coins[i]]
			}
		}
	}
	return dp[amount]
}
