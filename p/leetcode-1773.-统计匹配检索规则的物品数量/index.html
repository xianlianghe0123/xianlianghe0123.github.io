func countMatches(items [][]string, ruleKey string, ruleValue string) int {
	ans := 0
	for _, item := range items {
		value := ""
		switch ruleKey {
		case "type":
			value = item[0] 
		case "color":
			value = item[1]
		case "name":
			value = item[2]
		}
		if value == ruleValue {
			ans++
		}
	}
	return ans
}
