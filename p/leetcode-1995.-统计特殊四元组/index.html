func countQuadruplets(nums []int) int {
	cnt := 0
	for a := 0; a < len(nums)-3; a++ {
		for b := a + 1; b < len(nums)-2; b++ {
			for c := b + 1; c < len(nums)-1; c++ {
				for d := c + 1; d < len(nums); d++ {
					if nums[a]+nums[b]+nums[c] == nums[d] {
						cnt++
					}
				}
			}
		}
	}
	return cnt
}

func countQuadruplets(nums []int) int {
	cnt := 0
	hash := map[int]int{
		nums[len(nums)-1]: 1,
	}
	for c := len(nums) - 2; c >= 2; c-- {
		for b := c - 1; b >= 1; b-- {
			for a := b - 1; a >= 0; a-- {
				cnt += hash[nums[a]+nums[b]+nums[c]]
			}
		}
		hash[nums[c]]++
	}
	return cnt
}
