func chalkReplacer(chalk []int, k int) int {
	sum := 0
	for _, c := range chalk {
		sum += c
	}
	k %= sum
	for i, c := range chalk {
		k -= c
		if k < 0 {
			return i
		}
	}
	return 0
}

func chalkReplacer(chalk []int, k int) int {
	for i := 1; i < len(chalk); i++ {
		chalk[i] = chalk[i-1] + chalk[i]
	}
	k %= chalk[len(chalk)-1]
	return sort.Search(len(chalk), func(i int) bool {
		return chalk[i] > k
	})
}
