func deleteAndEarn(nums []int) int {
	maxNum := 0
	cnt := make(map[int]int)
	for _, num := range nums {
		cnt[num]++
		if maxNum < num {
			maxNum = num
		}
	}
	dp := make([]int, maxNum+1)
	dp[1] = cnt[1]
	for i := 2; i <= maxNum; i++ {
		dp[i] = dp[i-2] + cnt[i]*i
		if dp[i] < dp[i-1] {
			dp[i] = dp[i-1]
		}
	}
	return dp[maxNum]
}
