func minArray(numbers []int) int {
	min := math.MaxInt32
	for _, num := range numbers {
		if min > num {
			min = num
		}
	}
	return min
}

func minArray(numbers []int) int {
	if len(numbers) == 0 {
		return 0
	}
	l, r := 0, len(numbers)-1
	for l < r {
		mid := (l + r) / 2
		if numbers[mid] < numbers[r] {
			r = mid
		} else if numbers[mid] > numbers[r] {
			l = mid + 1
		} else {
			r--
		}
	}
	return numbers[l]
}
