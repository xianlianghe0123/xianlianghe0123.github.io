func lengthOfLongestSubstring(s string) int {
	ans := 0
	pos := make(map[byte]int)
	start := 0
	for i := 0; i < len(s); i++ {
		p, ok := pos[s[i]]
		if ok && start <= p {
			start = p + 1
		}
		pos[s[i]] = i
		if ans < i-start+1 {
			ans = i - start + 1
		}
	}
	return ans
}
