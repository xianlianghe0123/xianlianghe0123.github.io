func totalHammingDistance(nums []int) int {
	totalDistance := 0
	for i := 1 << 30; i > 0; i >>= 1 {
		ones := 0
		zeros := 0
		for _, num := range nums {
			if num&i == 0 {
				zeros++
			} else {
				ones++
			}
		}
		totalDistance += ones * zeros
	}
	return totalDistance
}
