func lowestCommonAncestor(root, p, q *TreeNode) *TreeNode {
    var dfs func(node *TreeNode, target *TreeNode, path []*TreeNode) []*TreeNode
    dfs = func(node *TreeNode, target *TreeNode, path []*TreeNode) []*TreeNode {
        if node == nil {
            return make([]*TreeNode, 0)
        }
        path = append(path, node)
        if node == target {
            return path
        }
        if path := dfs(node.Left, target, path); len(path) != 0 {
            return path
        }
        if path := dfs(node.Right, target, path); len(path) != 0 {
            return path
        }
        path = path[:len(path)-1]
        return make([]*TreeNode, 0)
    }
    pathP := dfs(root, p, make([]*TreeNode, 0, 2))
    pathQ := dfs(root, q, make([]*TreeNode, 0, 2))
    ans := (*TreeNode)(nil)
    for i := 0; i < len(pathP) && i < len(pathQ) && pathP[i] == pathQ[i]; i++ {
        ans = pathP[i]
    }
    return ans
}

func lowestCommonAncestor(root, p, q *TreeNode) *TreeNode {
    if root == nil {
        return nil
    }
    if root == p || root == q {
        return root
    }
    l := lowestCommonAncestor(root.Left, p, q)
    r := lowestCommonAncestor(root.Right, p, q)
    if l != nil && r != nil {
        return root
    }
    if l == nil {
        return r
    } else {
        return l
    }
}
