func sumRootToLeaf(root *TreeNode) int {
	var dfs func(node *TreeNode, prefix int) int
	dfs = func(node *TreeNode, prefix int) int {
		prefix <<= 1
		prefix |= node.Val
		if node.Left == nil && node.Right == nil {
			return prefix
		}
		sum := 0
		if node.Left != nil {
			sum += dfs(node.Left, prefix)
		}
		if node.Right != nil {
			sum += dfs(node.Right, prefix)
		}
		return sum
	}
	return dfs(root, 0)
}
