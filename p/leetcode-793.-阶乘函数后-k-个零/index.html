func zeros(n int) int {
	ans := 0
	for i := 5; i <= n; i *= 5 {
		ans += n / i
	}
	return ans
}

func preimageSizeFZF(k int) int {
	var binarySearch func(l, r int) bool
	binarySearch = func(l, r int) bool {
		if l > r {
			return false
		}
		mid := (l + r) / 2
		switch t := zeros(mid); {
		case t == k:
			fmt.Println(mid, t)
			return true
		case t > k:
			return binarySearch(l, mid-1)
		default:
			return binarySearch(mid+1, r)
		}
	}
	if binarySearch(0, 5*k) {
		return 5
	}
	return 0
}
