func soupServings(n int) float64 {
	if n >= 4475 {
		return 1
	}
	n = (n + 24) / 25
	dp := make([][]float64, n+1)
	for i := 0; i <= n; i++ {
		dp[i] = make([]float64, n+1)
		dp[0][i] = 1
	}
	dp[0][0] = 0.5
	for i := 1; i <= n; i++ {
		for j := 1; j <= n; j++ {
			for k := 4; k > 0; k-- {
				a := i - k
				if a < 0 {
					a = 0
				}
				b := j - (4 - k)
				if b < 0 {
					b = 0
				}
				dp[i][j] += dp[a][b]
			}
			dp[i][j] /= 4
		}
	}
	return dp[n][n]
}
