func fractionAddition(expression string) string {
	sum := &frac{
		numerator:   0,
		denominator: 1,
	}
	for i := 0; i < len(expression); {
		newFrac := new(frac)
		// ËØÜÂà´ÂàÜÂ≠ê
		j := i
		for ; j < len(expression) && (expression[j] >= '0' && expression[j] <= '9' || expression[j] == '+' || expression[j] == '-'); j++ {
		}
		newFrac.numerator, _ = strconv.Atoi(expression[i:j])
		i = j + 1 // Ë∑≥Ëøá/
		// ËØÜÂà´ÂàÜÊØç
		j = i
		for ; j < len(expression) && expression[j] >= '0' && expression[j] <= '9'; j++ {
		}
		newFrac.denominator, _ = strconv.Atoi(expression[i:j])
		i = j
		// Ê±ÇÂíå
		sum = add(sum, newFrac)
	}
	return fmt.Sprintf("%d/%d", sum.numerator, sum.denominator)
}

type frac struct {
	numerator   int
	denominator int
}

func add(f1, f2 *frac) *frac {
	ans := &frac{
		numerator:   f1.numerator*f2.denominator + f2.numerator*f1.denominator,
		denominator: f1.denominator * f2.denominator,
	}
	_gcd := gcd(ans.numerator, ans.denominator)
	ans.numerator /= _gcd
	ans.denominator /= _gcd
	return ans
}

func gcd(a int, b int) int {
	for b != 0 {
		a, b = b, a%b
	}
	if a < 0 {
		return -a
	}
	return a
}
