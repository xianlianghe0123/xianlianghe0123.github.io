func countMaxOrSubsets(nums []int) int {
	max := 0
	cnt := 0
	for bits := (1 << len(nums)) - 1; bits > 0; bits-- {
		or := 0
		for i := 0; i < len(nums); i++ {
			if bits&(1<<i) != 0 {
				or |= nums[i]
			}
		}
		if max < or {
			max = or
			cnt = 1
		} else if max == or {
			cnt++
		}
	}
	return cnt
}
