func hIndex(citations []int) int {
	for l, r := 0, len(citations)-1; ; {
		if l == r {
			if citations[l] >= len(citations)-l {
				return len(citations) - l
			} else {
				return 0
			}
		}
		mid := (l + r) / 2
		if citations[mid] < len(citations)-mid {
			l = mid + 1
		} else {
			r = mid
		}
	}
}
