func stoneGameIX(stones []int) bool {
	if len(stones) < 2 {
		return false
	}
	cnt := [...]int{0, 0, 0}
	for _, stone := range stones {
		cnt[stone%3]++
	}
	return (cnt[0]&1 == 1 && (cnt[2]-cnt[1] >= 3 || cnt[1]-cnt[2] >= 3)) || (cnt[0]&1 == 0 && cnt[1] > 0 && cnt[2] > 0)
}
