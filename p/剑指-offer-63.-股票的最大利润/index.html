func maxProfit(prices []int) int {
	ans := 0
	for i := range prices {
		for j := i + 1; j < len(prices); j++ {
			if ans < prices[j]-prices[i] {
				ans = prices[j] - prices[i]
			}
		}
	}
	return ans
}

func maxProfit(prices []int) int {
    min := int(math.MaxInt32)
    ans := 0
    for _, price := range prices {
        if min > price {
            min = price
        }
        if ans < price - min {
            ans = price - min
        }
    }
    return ans
}
