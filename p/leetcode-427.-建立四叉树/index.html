func construct(grid [][]int) *Node {
	var dfs func(row0, col0, row1, col1 int) *Node
	dfs = func(row0, col0, row1, col1 int) *Node {
		same := true
	Outer:
		for i := row0; i < row1; i++ {
			for j := col0; j < col1; j++ {
				if grid[i][j] != grid[row0][col0] {
					same = false
					break Outer
				}
			}
		}
		// ÂÖ®ÈÉ®Áõ∏ÂêåÔºåËÆæ‰∏∫Âè∂ÁªìÁÇπ
		if same {
			return &Node{
				Val:    grid[row0][col0] == 1,
				IsLeaf: true,
			}
		}
		// ÈÄíÂΩí
		midRow := (row0 + row1) / 2
		midCol := (col0 + col1) / 2
		return &Node{
			TopLeft:     dfs(row0, col0, midRow, midCol),
			TopRight:    dfs(row0, midCol, midRow, col1),
			BottomLeft:  dfs(midRow, col0, row1, midCol),
			BottomRight: dfs(midRow, midCol, row1, col1),
		}
	}
	return dfs(0, 0, len(grid), len(grid[0]))
}
