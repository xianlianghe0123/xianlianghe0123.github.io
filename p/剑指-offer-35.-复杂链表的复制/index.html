func copyRandomList(head *Node) *Node {
	m := make(map[*Node]*Node)
	newHead := new(Node)
	for node := newHead; head != nil; head, node = head.Next, node.Next {
		// è·å– Next æŒ‡é’ˆæŒ‡å‘ç»“ç‚¹
		t, ok := m[head]
		if !ok {
			t = &Node{
				Val: head.Val,
			}
			m[head] = t
		}
		// è·å– Random æŒ‡é’ˆæŒ‡å‘ç»“ç‚¹

				}

			}

		}
		node.Next = t
	}
	return newHead.Next
}

func copyRandomList(head *Node) *Node {
	if head == nil {
		return nil
	}
	// åˆ†è£‚
	for node := head; node != nil; node = node.Next.Next {
		node.Next = &Node{
			Val:    node.Val,
			Next:   node.Next,
			Random: node.Random,
		}
	}
	// random ä¿®æ­£
	for node := head; node != nil; node = node.Next.Next {
		if node.Random != nil {
			node.Next.Random = node.Random.Next
		}
	}
	// åˆ†ç¦»
	newHead := head.Next
	for oldNode, newNode := head, newHead; oldNode != nil; oldNode, newNode = oldNode.Next, newNode.Next {
		oldNode.Next = newNode.Next
		if newNode.Next != nil {
			newNode.Next = newNode.Next.Next
		}
	}
	return newHead
}
