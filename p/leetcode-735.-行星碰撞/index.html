func asteroidCollision(asteroids []int) []int {
	ans := make([]int, 0, len(asteroids))
	for _, a := range asteroids {
		if a > 0 {
			ans = append(ans, a)
			continue
		}
		for {
			// ä¸å‘ç”Ÿç¢°æ’ï¼Œ
			if len(ans) == 0 || ans[len(ans)-1] < 0 {
				ans = append(ans, a)
				break
			}
			sum := ans[len(ans)-1] + a
			if sum > 0 {
				break
			}
			if sum == 0 { // ä¸¤ä¸ªéƒ½çˆ†ç‚¸

				break
			}

		}
	}
	return ans
}
