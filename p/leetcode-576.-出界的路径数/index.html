func findPaths(m int, n int, maxMove int, startRow int, startColumn int) int {
	const MOD = 1e9 + 7
	dpCur := make([][]int, m)
	dpNext := make([][]int, m)
	for i := 0; i < m; i++ {
		dpCur[i] = make([]int, n)
		dpNext[i] = make([]int, n)
	}
	dpCur[startRow][startColumn] = 1
	res := 0
	for move := 0; move < maxMove; move++ {
		for row := 0; row < m; row++ {
			for col := 0; col < n; col++ {
				dpNext[row][col] = 0
				for _, offset := range [][]int{{-1, 0}, {1, 0}, {0, -1}, {0, 1}} {
                    // åœ¨è¾¹ç•Œå†…
					if row+offset[0] >= 0 && row+offset[0] < m && col+offset[1] >= 0 && col+offset[1] < n {
						dpNext[row][col] = (dpNext[row][col] + dpCur[row+offset[0]][col+offset[1]]) % MOD
					} else {
						res = (res + dpCur[row][col]) % MOD
					}
				}
			}
		}
		dpCur, dpNext = dpNext, dpCur
		fmt.Println(dpCur, res)
	}
	return res
}
