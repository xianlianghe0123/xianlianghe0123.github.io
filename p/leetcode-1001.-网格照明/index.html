func gridIllumination(n int, lamps [][]int, queries [][]int) []int {
	lampCoord := make(map[[2]int]struct{})
	rows := make(map[int]int)         // Ë°å
	cols := make(map[int]int)         // Âàó
	diagonal := make(map[int]int)     // ÂØπËßíÁ∫ø
	backDiagonal := make(map[int]int) // ÂèçÂØπËßíÁ∫ø
	for _, lamp := range lamps {
		// Âêå‰∏Ä‰∏™ÂçïÂÖÉÊ†ºÈáåÁöÑÁÅØÂΩì‰Ωú‰∏ÄÁõèÁÅØÂ§ÑÁêÜ
		if _, ok := lampCoord[[2]int{lamp[0], lamp[1]}]; ok {
			continue
		}
		lampCoord[[2]int{lamp[0], lamp[1]}] = struct{}{}
		rows[lamp[0]]++
		cols[lamp[1]]++
		diagonal[lamp[1]-lamp[0]]++
		backDiagonal[lamp[1]+lamp[0]]++
	}
	ans := make([]int, 0, len(queries))
	for _, query := range queries {
		if rows[query[0]] > 0 || cols[query[1]] > 0 || diagonal[query[1]-query[0]] > 0 || backDiagonal[query[1]+query[0]] > 0 {
			ans = append(ans, 1)
		} else {
			ans = append(ans, 0)
		}
		// ÂÖ≥Èó≠‰πùÂÆ´Ê†ºÈáåÁöÑÁÅØ
		for i := query[0] - 1; i <= query[0]+1; i++ {
			for j := query[1] - 1; j <= query[1]+1; j++ {
				if i < 0 || i >= n || j < 0 || j >= n {
					continue
				}
				lamp := [2]int{i, j}
				if _, ok := lampCoord[lamp]; !ok {
					continue
				}
				delete(lampCoord, lamp)
				rows[lamp[0]]--
				cols[lamp[1]]--
				diagonal[lamp[1]-lamp[0]]--
				backDiagonal[lamp[1]+lamp[0]]--
			}
		}
	}
	return ans
}
