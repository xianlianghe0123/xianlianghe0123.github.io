func isPalindrome(head *ListNode) bool {
	str := make([]int, 0, 1e5)
	for ; head != nil; head = head.Next {
		str = append(str, head.Val)
	}
	for i := len(str)/2 - 1; i >= 0; i-- {
		if str[i] != str[len(str)-1-i] {
			return false
		}
	}
	return true
}

	return true
}
