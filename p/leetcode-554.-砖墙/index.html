func leastBricks(wall [][]int) int {
	gapCnt := make(map[int]int)
	for _, bricks := range wall {
		gap := 0
		l := len(bricks) - 1
		for i := 0; i < l; i++ {
			gap += bricks[i]
			gapCnt[gap]++
		}
	}
	maxGaps := 0
	for _, v := range gapCnt {
		if maxGaps < v {
			maxGaps = v
		}
	}
	return len(wall) - maxGaps
}
