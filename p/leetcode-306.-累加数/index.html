func isAdditiveNumber(num string) bool {
	for i := 1; i <= len(num)/2; i++ {
		for j := i + 1; j < len(num); j++ {
			// å‰å¯¼ 0 è·³è¿‡
			if j-i > 1 && num[i] == '0' {
				break
			}
			num1 := num[:i]
			num2 := num[i:j]
			// è®¡ç®—æ˜¯å¦æ˜¯ç´¯åŠ å’Œ
			for k := j; k <= len(num); {
				sum := add(num1, num2)
				if !strings.HasPrefix(num[k:], sum) {
					break
				}
				k += len(sum)
				if k == len(num) {
					return true
				}
				num1, num2 = num2, sum
			}
		}
	}
	return false
}

func add(s1 string, s2 string) string {
	if len(s1) < len(s2) {
		s1, s2 = s2, s1
	}
	result := make([]byte, len(s1)+1)
	c := byte(0)
	for i, j, k := len(s1)-1, len(s2)-1, len(result)-1; i >= 0 || j >= 0; i, j, k = i-1, j-1, k-1 {
		switch {
		case j < 0:
			result[k] = s1[i] + c
		default:
			result[k] = s1[i] + s2[j] - byte('0') + c
		}
		c = 0
		if result[k] > '9' {
			result[k] -= 10
			c = 1
		}
	}
	if c == 1 {
		result[0] = '1'
		return string(result)
	}
	return string(result[1:])
}
