func cuttingRope(n int) int {
	if n <= 3 {
		return n - 1
	}
	const MOD = int(1e9 + 7)
	fastPower := func(x, n int) int {
		power := 1
		for ; n > 0; n >>= 1 {
			if n&1 == 1 {
				power = (power * x) % MOD
			}
			x = (x * x) % MOD
		}
		return power
	}
	switch threes, remain := n/3, n%3; remain {
	case 0:
		return fastPower(3, threes)
	case 1:
		return (fastPower(3, threes-1) * 4) % MOD
	case 2:
		return (fastPower(3, threes) * 2) % MOD
	}
	return 0
}
