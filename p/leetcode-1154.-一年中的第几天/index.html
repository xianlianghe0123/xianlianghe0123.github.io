func dayOfYear(date string) int {
	monthDays := []int{31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}
	year, _ := strconv.Atoi(date[:4])
	month, _ := strconv.Atoi(date[5:7])
	day, _ := strconv.Atoi(date[8:])
	// é—°å¹´äºŒæœˆå¤šä¸€å¤©
	if (year%400 == 0) || (year%4 == 0 && year%100 != 0) {
		monthDays[1]++
	}
	days := day
	for i := month - 2; i >= 0; i-- {
		days += monthDays[i]
	}
	return days
}
