func isCousins(root *TreeNode, x int, y int) bool {
	parents := make([]int, 0, 2)
	heights := make([]int, 0, 2)
	var dfs func(*TreeNode, int, int)
	dfs = func(node *TreeNode, parent, height int) {
		if node == nil {
			return
		}
		if node.Val == x || node.Val == y {
			heights = append(heights, height)
			parents = append(parents, parent)
		}
		dfs(node.Left, node.Val, height+1)
		dfs(node.Right, node.Val, height+1)
	}
	dfs(root, 0, 1)
	if parents[0] != parents[1] && heights[0] == heights[1] {
		return true
	}
	return false
}
