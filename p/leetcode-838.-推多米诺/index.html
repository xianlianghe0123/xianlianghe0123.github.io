func pushDominoes(dominoes string) string {
	dominoesBytes := []byte(dominoes)
	// ÂàùÂßãÁä∂ÊÄÅ
	l, r := 0, 0
	for r = range dominoesBytes {
		if r == len(dominoesBytes)-1 || dominoesBytes[r] != '.' {
			if dominoesBytes[l] == 'R' && dominoesBytes[r] == 'L' {
				for i := 1; l+i < r-i; i++ {
					dominoesBytes[l+i] = dominoesBytes[l]
					dominoesBytes[r-i] = dominoesBytes[r]
				}
			} else if dominoesBytes[l] == '.' && dominoesBytes[r] == 'L' {
				for i := l; i < r; i++ {
					dominoesBytes[i] = 'L'
				}
			} else if dominoesBytes[l] == 'R' && dominoesBytes[r] == '.' {
				for i := l + 1; i <= r; i++ {
					dominoesBytes[i] = 'R'
				}
			} else if dominoesBytes[l] == dominoesBytes[r] {
				for i := l + 1; i < r; i++ {
					dominoesBytes[i] = dominoesBytes[l]
				}
			}
			l = r
		}
	}
	return string(dominoesBytes)
}
