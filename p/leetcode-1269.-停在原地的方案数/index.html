func numWays(steps int, arrLen int) int {
	const MOD = 1_000_000_007
	far := steps + 1
	if far > arrLen {
		far = arrLen
	}
	dp := make([]int, far) // ÁßªÂä®0Ê¨°
	dp[0] = 1              // ÂàùÂßãÂú®0
	for i := 1; i < steps; i++ {
		last := dp[0]
		for j := 0; j < far; j++ {
			t := dp[j]
			if j == 0 { // Â∑¶ËæπË∂äÁïå
				t = (t + dp[j+1]) % MOD
			} else if j == far-1 { // Âè≥ËæπË∂äÁïå

			}

		}
	}

	if arrLen > 1 {
		res = (res + dp[1]) % MOD
	}
	return res
}
