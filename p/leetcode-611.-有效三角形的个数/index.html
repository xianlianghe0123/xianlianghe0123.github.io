func triangleNumber(nums []int) int {
	sort.Ints(nums)
	res := 0
	for i := 0; i < len(nums)-2; i++ {
		if nums[i] == 0 {
			continue
		}
		for j := i + 1; j < len(nums)-1; j++ {
			pos := sort.SearchInts(nums, nums[i]+nums[j])
			pos--
			if pos-j > 0 {
				res += pos - j
			}
		}
	}
	return res
}

	sort.Ints(nums)
	res := 0
	for i := 0; i < len(nums)-2; i++ {
		if nums[i] == 0 {
			continue
		}
		for j, k := i+1, i+1; j < len(nums)-1; j++ {
			for ; k+1 < len(nums) && nums[k+1] < nums[i]+nums[j]; k++ {
			}
			res += k - j
		}
	}
	return res
}
