func numTilings(n int) int {
	const Mod = 1e9 + 7
	dp0 := []int{0, 0, 0, 1}
	dp1 := make([]int, 4)
	for i := 0; i < n; i++ {
		dp1[0] = dp0[3]
		dp1[1] = (dp0[0] + dp0[2]) % Mod
		dp1[2] = (dp0[0] + dp0[1]) % Mod
		dp1[3] = (dp0[0] + dp0[1] + dp0[2] + dp0[3]) % Mod
		dp0, dp1 = dp1, dp0
	}
	return dp0[3]
}
