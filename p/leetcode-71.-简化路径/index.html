func simplifyPath(path string) string {
	stack := make([]string, 0, 10)
	begin := 0
	for i := 0; i <= len(path); i++ {
		// ç›®å½•åˆ†éš”ç¬¦ æˆ– ç»“å°¾
		if i == len(path) || path[i] == '/' {
			// å­˜åœ¨æœ‰æ•ˆ
			if i > begin {
				switch path[begin:i] {
				case ".":
				case "..":
					if len(stack) > 0 {
						stack = stack[:len(stack)-1]
					}
				default:
					stack = append(stack, path[begin:i])
				}
			}
			begin = i + 1
		}
	}
	return "/" + strings.Join(stack, "/")
}
