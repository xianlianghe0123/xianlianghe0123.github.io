func subdomainVisits(cpdomains []string) []string {
	cnt := make(map[string]int)
	for _, cp := range cpdomains {
		// è§£æè®¿é—®æ¬¡æ•°
		i := strings.IndexByte(cp, ' ')
		rep, _ := strconv.Atoi(cp[:i])
		cp = cp[i+1:]
		// æ›´æ–°æ‰€æœ‰åŸŸåçš„è®¿é—®æ¬¡æ•°
		cnt[cp] += rep
		for i := range cp {
			if cp[i] == '.' {
				cnt[cp[i+1:]] += rep
			}
		}
	}
	ans := make([]string, 0, len(cnt))
	for k, v := range cnt {
		ans = append(ans, fmt.Sprintf("%d %s", v, k))
	}
	return ans
}
