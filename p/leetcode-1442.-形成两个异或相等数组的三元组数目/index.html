func countTriplets(arr []int) int {
	cnt := 0
	for i := 0; i < len(arr); i++ {
		a := 0
		for j := i + 1; j < len(arr); j++ {
			a ^= arr[j-1]
			b := 0
			for k := j; k < len(arr); k++ {
				b ^= arr[k]
				if a == b {
					cnt++
				}
			}
		}
	}
	return cnt
}

func countTriplets(arr []int) int {
	cnt := 0
	for i := 0; i < len(arr); i++ {
		a := arr[i]
		for j := i + 1; j < len(arr); j++ {
			a ^= arr[j]
			if a == 0 {
				cnt += j - i
			}
		}
	}
	return cnt
}
