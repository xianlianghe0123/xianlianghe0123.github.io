func checkRecord(s string) bool {
	absent := 0
	late := 0
	for i := range s {
		switch s[i] {
		case 'A':
			absent++
			late = 0
		case 'L':
			late++
		case 'P':
			late = 0
		}
		if absent >= 2 || late >= 3 {
			return false
		}
	}
	return true
}
