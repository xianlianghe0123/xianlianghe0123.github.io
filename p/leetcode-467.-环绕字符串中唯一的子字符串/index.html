func findSubstringInWraproundString(p string) int {
	maxLength := make([]int, 26)
	length := 1 // ËÆ∞ÂΩï‰ª•ÂΩìÂâçÈÅçÂéÜÂà∞ÁöÑÂ≠óÁ¨¶‰∏∫ÁªìÂ∞æÁöÑÔºåÊª°Ë∂≥Êù°‰ª∂ÁöÑÊúÄÈïøÂ≠ê‰∏≤ÈïøÂ∫¶
	maxLength[p[0]-'a'] = 1
	for i := 1; i < len(p); i++ {
		// Âà§Êñ≠ÊòØÂê¶ËøûÁª≠
		if (p[i]+26-p[i-1])%26 == 1 {
			length++
		} else {
			length = 1
		}
		if maxLength[p[i]-'a'] < length {
			maxLength[p[i]-'a'] = length
		}
	}
	sum := 0
	for _, l := range maxLength {
		sum += l
	}
	return sum
}
