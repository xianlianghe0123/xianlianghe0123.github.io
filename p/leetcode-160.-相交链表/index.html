func getIntersectionNode(headA, headB *ListNode) *ListNode {
	hash := make(map[*ListNode]bool)
	for node := headA; node != nil; node = node.Next {
		hash[node] = true
	}
	for node := headB; node != nil; node = node.Next {
		if hash[node] {
			return node
		}
	}
	return nil
}

func getIntersectionNode(headA, headB *ListNode) *ListNode {
	// ËÆ∞Èïø
	lenA := 0
	for node := headA; node != nil; node = node.Next {
		lenA++
	}
	lenB := 0
	for node := headB; node != nil; node = node.Next {
		lenB++
	}
	nodeA := headA
	nodeB := headB
	// ÂØπÈΩê
	if lenA > lenB {
		for i := 0; i < lenA-lenB; i++ {
			nodeA = nodeA.Next
		}
	} else {
		for i := 0; i < lenB-lenA; i++ {
			nodeB = nodeB.Next
		}
	}
	// ÊØîËæÉ
	for ; nodeA != nil; nodeA, nodeB = nodeA.Next, nodeB.Next {
		if nodeA == nodeB {
			return nodeA
		}
	}
	return nil
}
