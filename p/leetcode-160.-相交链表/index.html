func getIntersectionNode(headA, headB *ListNode) *ListNode {
	hash := make(map[*ListNode]bool)
	for node := headA; node != nil; node = node.Next {
		hash[node] = true
	}
	for node := headB; node != nil; node = node.Next {
		if hash[node] {
			return node
		}
	}
	return nil
}
