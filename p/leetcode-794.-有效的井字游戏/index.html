func validTicTacToe(board []string) bool {
	n := len(board)
	xNum, oNum := 0, 0 // Ê£ãÂ≠êÊï∞ÈáèÁªüËÆ°
	for _, line := range board {
		for j := range board {
			switch line[j] {
			case 'X':
				xNum++
			case 'O':
				oNum++
			}
		}
	}
	// Êï∞Èáè‰∏çÁ¨¶ÂêàÈ¢òËÆæ
	if xNum < oNum || xNum-oNum > 1 {
		return false
	}
	xWin, oWin := false, false // ÊòØÂê¶Êª°Ë∂≥Ëé∑ËÉúÊù°‰ª∂
	// ÊòØÂê¶ÊúâÁõ∏ÂêåË°åÔºåÂàóÔºåÂØπËßíÁ∫øÔºåËøõË°å‰øÆÊ≠£
	win := func(same bool, ch byte) {
		if !same {
			return
		}
		switch ch {
		case 'X':
			xWin = true
		case 'O':
			oWin = true
		}
	}
	for i := 0; i < n; i++ {
		rowSame, colSame := true, true
		for j := 1; j < n; j++ {
			// Ë°åÊâ´Êèè
			if board[i][j] != board[i][0] {
				rowSame = false
			}
			// ÂàóÊâ´Êèè
			if board[j][i] != board[0][i] {
				colSame = false
			}
		}
		win(rowSame, board[i][0])
		win(colSame, board[0][i])
	}
	// ÂØπËßíÁ∫øÊâ´Êèè
	forwordSame, backSame := true, true
	for i := n - 1; i > 0; i-- {
		if board[i][i] != board[0][0] {
			forwordSame = false
		}
		if board[i][n-1-i] != board[0][n-1] {
			backSame = false
		}
	}
	win(forwordSame, board[0][0])
	win(backSame, board[0][n-1])
	// ‰∏çÊª°Ë∂≥È¢òËÆæÊù°‰ª∂
	if (xWin && oWin) || (xWin && xNum <= oNum) || (oWin && xNum > oNum) {
		return false
	}
	return true
}
