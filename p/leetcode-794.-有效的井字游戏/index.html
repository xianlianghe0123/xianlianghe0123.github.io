func validTicTacToe(board []string) bool {
	n := len(board)
	xNum, oNum := 0, 0 // æ£‹å­æ•°é‡ç»Ÿè®¡
	for _, line := range board {
		for j := range board {
			switch line[j] {
			case 'X':
				xNum++
			case 'O':
				oNum++
			}
		}
	}
	// æ•°é‡ä¸ç¬¦åˆé¢˜è®¾
	if xNum < oNum || xNum-oNum > 1 {
		return false
	}
	xWin, oWin := false, false // æ˜¯å¦æ»¡è¶³è·èƒœæ¡ä»¶
	// æ˜¯å¦æœ‰ç›¸åŒè¡Œï¼Œåˆ—ï¼Œå¯¹è§’çº¿ï¼Œè¿›è¡Œä¿®æ­£
	win := func(same bool, ch byte) {
		if !same {
			return
		}
		switch ch {
		case 'X':
			xWin = true
		case 'O':
			oWin = true
		}
	}
	for i := 0; i < n; i++ {
		rowSame, colSame := true, true
		for j := 1; j < n; j++ {
			// è¡Œæ‰«æ
			if board[i][j] != board[i][0] {
				rowSame = false
			}
			// åˆ—æ‰«æ
			if board[j][i] != board[0][i] {
				colSame = false
			}
		}
		win(rowSame, board[i][0])
		win(colSame, board[0][i])
	}
	// å¯¹è§’çº¿æ‰«æ
	forwordSame, backSame := true, true
	for i := n - 1; i > 0; i-- {
		if board[i][i] != board[0][0] {
			forwordSame = false
		}
		if board[i][n-1-i] != board[0][n-1] {
			backSame = false
		}
	}
	win(forwordSame, board[0][0])
	win(backSame, board[0][n-1])
	// ä¸æ»¡è¶³é¢˜è®¾æ¡ä»¶
	if (xWin && oWin) || (xWin && xNum <= oNum) || (oWin && xNum > oNum) {
		return false
	}
	return true
}
