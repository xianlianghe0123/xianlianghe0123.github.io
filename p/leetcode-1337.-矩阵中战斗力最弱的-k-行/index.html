func kWeakestRows(mat [][]int, k int) []int {
	type Node struct {
		Row     int
		Soldier int
	}
	nodes := make([]*Node, 0, len(mat))
	for i := range mat {
		node := &Node{
			Row:     i,
			Soldier: 0,
		}
		for _, p := range mat[i] {
			if p == 1 {
				node.Soldier++
			} else {
				break
			}
		}
		nodes = append(nodes, node)
	}
	sort.Slice(nodes, func(i, j int) bool {
		if nodes[i].Soldier < nodes[j].Soldier {
			return true
		} else if nodes[i].Soldier > nodes[j].Soldier {
			return false
		} else {
			return nodes[i].Row < nodes[j].Row
		}
	})
	res := make([]int, 0, k)
	for i := 0; i < k; i++ {
		res = append(res, nodes[i].Row)
	}
	return res
}

func kWeakestRows(mat [][]int, k int) []int {
	type Node struct {
		Row     int
		Soldier int
	}
	nodes := make([]*Node, 0, len(mat))
	for i := range mat {
		nodes = append(nodes, &Node{
			Row: i,
			Soldier: sort.Search(len(mat[i]), func(j int) bool {
				return mat[i][j] == 0
			})},
		)
	}
	sort.Slice(nodes, func(i, j int) bool {
		if nodes[i].Soldier < nodes[j].Soldier {
			return true
		} else if nodes[i].Soldier > nodes[j].Soldier {
			return false
		} else {
			return nodes[i].Row < nodes[j].Row
		}
	})
	res := make([]int, 0, k)
	for i := 0; i < k; i++ {
		res = append(res, nodes[i].Row)
	}
	return res
}
