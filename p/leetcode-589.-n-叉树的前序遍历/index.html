func preorder(root *Node) []int {
	if root == nil {
		return nil
	}
	elems := []int{root.Val}
	for _, child := range root.Children {
		elems = append(elems, preorder(child)...)
	}
	return elems
}

func preorder(root *Node) []int {
	if root == nil {
		return nil
	}
	ans := make([]int, 0, 1)
	stack := []*Node{root}
	for len(stack) > 0 {
		cur := stack[len(stack)-1] // ÂΩìÂâçÈÅçÂéÜÁöÑÁªìÁÇπ
		stack = stack[:len(stack)-1] // Âá∫Ê†à
		ans = append(ans, cur.Val)
        // Â≠êÁªìÁÇπ‰ªéÂè≥ÂæÄÂ∑¶ÂÖ•Ê†à
		for i := len(cur.Children) - 1; i >= 0; i-- {
			if cur.Children[i] == nil {
				continue
			}
			stack = append(stack, cur.Children[i])
		}
	}
	return ans
}
