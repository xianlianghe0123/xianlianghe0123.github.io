func findAndReplacePattern(words []string, pattern string) []string {
	ans := make([]string, 0, len(words))
	for _, word := range words {
		p2w := make(map[byte]byte) // è®°å½•æ˜ å°„å…³ç³»
		cnt := make([]int, 26)
		valid := true
		for i := range pattern {
			char, ok := p2w[pattern[i]]
			// è®°å½•æ˜ å°„å…³ç³»
			if !ok {
				p2w[pattern[i]] = word[i]

				cnt[word[i]-'a']++
				if cnt[word[i]-'a'] > 1 {
					valid = false
					break
				}
				continue
			}
			if char != word[i] {
				valid = false
				break
			}
		}
		if !valid {
			continue
		}
		ans = append(ans, word)
	}
	return ans
}
