func shortestToChar(s string, c byte) []int {
	distance := make([]int, len(s))
	l, r := -1, 0
	for i := range s {
		if i < len(s)-1 && s[i] != c {
			continue
		}
		l, r = r, i
		for j := l; j <= r; j++ {
			distance[j] = math.MaxInt
			if s[l] == c && distance[j] > j-l {
				distance[j] = j - l
			}
			if s[r] == c && distance[j] > r-j {
				distance[j] = r - j
			}
		}
	}
	return distance
}
