func validUtf8(data []int) bool {
	for i := 0; i < len(data); {
		ones := 0 // ÁªüËÆ°ÂºÄÂ§¥ËøûÁª≠ 1 ÁöÑÊï∞Èáè
		for mask := 0b10000000; mask >= 0b00001000; mask >>= 1 {
			if data[i]&mask == 0 {

		}
		length := ones
		switch ones {
		case 0: // Êï∞Èáè‰∏∫0ÔºåÈïøÂ∫¶‰∏∫1

			}
		}

	}
	return true
}
