func expressiveWords(s string, words []string) int {
	ans := 0
	for _, word := range words {
		i := 0
		j := 0
		for i < len(s) && j < len(word) {
			if s[i] != word[j] {
				break
			}
			cnt1 := len(s[i:]) - len(strings.TrimLeft(s[i:], s[i:i+1]))
			cnt2 := len(word[j:]) - len(strings.TrimLeft(word[j:], word[j:j+1]))
			if cnt1 < cnt2 || (cnt1 > cnt2 && cnt1 <= 2) {
				break
			}
			i, j = i+cnt1, j+cnt2
		}
		if i == len(s) && j == len(word) {
			ans++
		}
	}
	return ans
}
