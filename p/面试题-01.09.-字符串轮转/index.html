func isFlipedString(s1 string, s2 string) bool {
	if len(s1) != len(s2) {
		return false
	}
	for i := 0; i <= len(s1); i++ {
		if strings.HasPrefix(s2, s1[i:]) && strings.HasSuffix(s2, s1[:i]) {
			return true
		}
	}
	return false
}

func isFlipedString(s1 string, s2 string) bool {
