type MaxQueue struct {
	queue    *list.List
	maxQueue *list.List
}

func Constructor() MaxQueue {
	return MaxQueue{
		queue:    list.New(),
		maxQueue: list.New(),
	}
}

func (this *MaxQueue) Max_value() int {
	if this.maxQueue.Len() == 0 {
		return -1
	}
	return this.maxQueue.Front().Value.(int)
}

func (this *MaxQueue) Push_back(value int) {
	this.queue.PushBack(value)
	for this.maxQueue.Len() > 0 && this.maxQueue.Back().Value.(int) < value {
		this.maxQueue.Remove(this.maxQueue.Back())
	}
	this.maxQueue.PushBack(value)
}

func (this *MaxQueue) Pop_front() int {
	if this.queue.Len() == 0 {
		return -1
	}
	front := this.queue.Front().Value.(int)
	this.queue.Remove(this.queue.Front())
	if this.maxQueue.Front().Value.(int) == front {
		this.maxQueue.Remove(this.maxQueue.Front())
	}
	return front
}
