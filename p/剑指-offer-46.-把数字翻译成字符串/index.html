func translateNum(num int) int {
	s := strconv.FormatInt(int64(num), 10)
	dp := make([]int, len(s))
	dp[0] = 1
	for i := 0; i < len(s)-1; i++ {
		dp[i+1] = dp[i]
		if s[i] != '0' && (s[i]-'0')*10+(s[i+1]-'0') < 26 {
			if i-1 >= 0 {
				dp[i+1] += dp[i-1]
			} else {
				dp[i+1]++
			}
		}
	}
	return dp[len(s)-1]
}
