func minMoves2(nums []int) int {
	min := math.MaxInt
	for _, num := range nums {
		if min > num {
			min = num
		}
	}
	moves := 0
	for _, num := range nums {
		moves += num - min
	}
	sort.Ints(nums)
	minMoves := moves
	for i := 1; i < len(nums); i++ {
		moves += (nums[i] - nums[i-1]) * (2*i - len(nums))
		if minMoves > moves {
			minMoves = moves
		}
	}
	return minMoves
}
