func leafSimilar(root1 *TreeNode, root2 *TreeNode) bool {
	var dfs func(node *TreeNode, leafs *[]int)
	dfs = func(node *TreeNode, leafs *[]int) {
		fmt.Println(node.Val)
		if node.Left == nil && node.Right == nil {
			*leafs = append(*leafs, node.Val)
		} else {
			if node.Left != nil {
				dfs(node.Left, leafs)
			}
			if node.Right != nil {
				dfs(node.Right, leafs)
			}
		}
	}
	leafs1 := make([]int, 0, 100)
	dfs(root1, &leafs1)
	leafs2 := make([]int, 0, 100)
	dfs(root2, &leafs2)
	if len(leafs1) != len(leafs2) {
		return false
	}
	for i := range leafs1 {
		if leafs1[i] != leafs2[i] {
			return false
		}
	}
	return true
}
