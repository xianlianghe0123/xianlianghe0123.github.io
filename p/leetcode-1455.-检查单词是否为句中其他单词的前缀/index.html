func isPrefixOfWord(sentence string, searchWord string) int {
	idx := 1
	for start, end := 0, 0; end <= len(sentence); end++ {
		if end == len(sentence) || sentence[end] == ' ' {
			if strings.HasPrefix(sentence[start:end], searchWord) {
				return idx
			}
			idx++
			start = end + 1
		}
	}
	return -1
}
