func networkBecomesIdle(edges [][]int, patience []int) int {
	// ÊûÑÈÄ†ÈÇªÊé•ÈìæË°®
	next := make([][]int, len(patience))
	for _, edge := range edges {
		next[edge[0]] = append(next[edge[0]], edge[1])
		next[edge[1]] = append(next[edge[1]], edge[0])
	}
	minTime := make([]int, len(patience)) // ËÆ∞ÂΩïÊúÄÁü≠Êó∂Èó¥
	visit := make([]bool, len(patience))
	queue := list.New()
	queue.PushBack(0)
	visit[0] = true
	for time := 0; queue.Len() > 0; time++ {
		// Âêå‰∏ÄÊó∂Èó¥Âà∞Ëææ
		for i := queue.Len(); i > 0; i-- {
			cur := queue.Remove(queue.Front()).(int)
			minTime[cur] = time // ËÆ∞ÂΩïÊúÄÁü≠Êó∂Èó¥
			// ÈÅçÂéÜ‰∏ã‰∏™ÁªìÁÇπ
			for _, n := range next[cur] {
				if !visit[n] {
					queue.PushBack(n)
					visit[n] = true
				}
			}
		}
	}
	ans := 0
	for i, m := range minTime {
		t := m << 1
		// ÊúâÈáçÂèëÊ∂àÊÅØÔºåÊé•Êî∂Âà∞ÊúÄÂêé‰∏ÄÊ¨°Ê∂àÊÅØÂêéÁ©∫Èó≤
		if patience[i] < (m << 1) {
			t = (m << 1) + (m<<1-1)/patience[i]*patience[i]
		}
		if ans < t {
			ans = t
		}
	}
	return ans + 1
}
