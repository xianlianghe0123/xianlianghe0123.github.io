func reorderedPowerOf2(n int) bool {
	proc := func(num int) string {
		t := make([]byte, 0, 10)
		for ; num > 0; num /= 10 {
			t = append(t, byte(num%10)+'0')
		}
		sort.Slice(t, func(i, j int) bool {
			return t[i] < t[j]
		})
		return string(t)
	}
	exist := make(map[string]struct{})
	for i := 1; i < n*10; i <<= 1 {
		exist[proc(i)] = struct{}{}
	}
	_, ok := exist[proc(n)]
	return ok
}
