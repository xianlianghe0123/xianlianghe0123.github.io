		}
        // ÈÄíÂΩíÊûÑÂª∫

			TopRight:    dfs(n1.TopRight, n2.TopRight, n1.Val, n2.Val),
			BottomLeft:  dfs(n1.BottomLeft, n2.BottomLeft, n1.Val, n2.Val),
			BottomRight: dfs(n1.BottomRight, n2.BottomRight, n1.Val, n2.Val),
		}

			}

	}

}
