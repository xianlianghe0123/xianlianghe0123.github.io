func findPoisonedDuration(timeSeries []int, duration int) int {
	time := duration
	for i := 1; i < len(timeSeries); i++ {
		if timeSeries[i]-timeSeries[i-1] >= duration {
			time += duration
		} else {
			time += timeSeries[i] - timeSeries[i-1]
		}
	}
	return time
}
