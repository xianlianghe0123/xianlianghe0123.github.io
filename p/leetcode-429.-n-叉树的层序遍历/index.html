func levelOrder(root *Node) [][]int {
	if root == nil {
		return nil
	}
	ans := make([][]int, 0, 1)
	queue := list.New()
	queue.PushBack(root)
	layerCnt := 1 // è®°å½•ä¸‹ä¸€å±‚çš„ç»“ç‚¹æ•°ï¼Œæå‰å¼€è¾Ÿç©ºé—´
	for queue.Len() > 0 {

		layerCnt = 0                           // å½’é›¶ï¼Œè®¡ç®—ä¸‹ä¸€å±‚çš„ç»“ç‚¹æ•°

		for i := queue.Len(); i > 0; i-- {
			node := queue.Remove(queue.Front()).(*Node)

			layerCnt += len(node.Children) // ä¸‹ä¸€å±‚ç»“ç‚¹æ•°ç´¯åŠ

	}
	return ans
}
