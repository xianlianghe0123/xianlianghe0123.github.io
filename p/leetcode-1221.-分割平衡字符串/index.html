func balancedStringSplit(s string) int {
	split := 0
	balance := 0
	for i := 0; i < len(s); i++ {
		switch s[i] {
		case 'L':
			balance--
		case 'R':
			balance++
		}
		if balance == 0 {
			split++
		}
	}
	return split
}
