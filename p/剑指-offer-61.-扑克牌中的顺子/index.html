func isStraight(nums []int) bool {
	sort.Ints(nums)
	cnt := 0
	for i := 0; i < len(nums); i++ {
		if nums[i] == 0 {
			cnt++
			continue
		}
		if i-1 >= 0 && nums[i-1] != 0 {
			diff := nums[i] - nums[i-1]
			if diff < 1 {
				return false
			}
			cnt -= diff - 1
		}
	}
	return cnt >= 0
}
