func sequenceReconstruction(nums []int, sequences [][]int) bool {
	inDegree := make([]int, len(nums)+1) // ÂÖ•Â∫¶
	next := make([][]int, len(nums)+1)   // ÈÇªÊé•ÈìæË°®
	for _, seq := range sequences {
		for i := 1; i < len(seq); i++ {
			next[seq[i-1]] = append(next[seq[i-1]], seq[i])
			inDegree[seq[i]]++
		}
	}
	zeroInDegreeNums := make([]int, 0, 1)
	for _, num := range nums {
		if inDegree[num] == 0 {
			zeroInDegreeNums = append(zeroInDegreeNums, num)
		}
	}
	for len(zeroInDegreeNums) > 0 {
		// ÊãìÊâëÊéíÂ∫èÁªìÊûú‰∏çÂîØ‰∏Ä
		if len(zeroInDegreeNums) != 1 {
			return false
		}
		n := zeroInDegreeNums[0]
		zeroInDegreeNums = zeroInDegreeNums[1:]
		// ÊâÄÊúâÂêéÁªßËäÇÁÇπÂØπÂ∫îÂÖ•Â∫¶ÂéªÈô§Ôºå
		// ÂØªÊâæÂÖ•Â∫¶‰∏∫ 0 ÁöÑËäÇÁÇπ
		for _, num := range next[n] {
			inDegree[num]--
			if inDegree[num] == 0 {
				zeroInDegreeNums = append(zeroInDegreeNums, num)
			}
		}
	}
	return true
}
