func xorQueries(arr []int, queries [][]int) []int {
	res := make([]int, 0, len(queries))
	for _, query := range queries {
		t := 0
		for i := query[0]; i <= query[1]; i++ {
			t ^= arr[i]
		}
		res = append(res, t)
	}
	return res
}

func xorQueries(arr []int, queries [][]int) []int {
	xor := make([]int, 0, len(arr)+1)
	xor = append(xor, 0)
	for i := 0; i < len(arr); i++ {
		xor = append(xor, xor[i]^arr[i])
	}
	res := make([]int, 0, len(queries))
	for _, query := range queries {
		res = append(res, xor[query[0]]^xor[query[1]+1])
	}
	return res
}
