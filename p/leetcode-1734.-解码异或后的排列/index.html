func decode(encoded []int) []int {
	perm := make([]int, 0, len(encoded)+1)
	tmp := 0
	for i := 1; i <= cap(perm); i++ {
		tmp ^= i
	}
	for i := 1; i < len(encoded); i += 2 {
		tmp ^= encoded[i]
	}
	perm = append(perm, tmp)
	for i := range encoded {
		perm = append(perm, perm[i]^encoded[i])
	}
	return perm
}
