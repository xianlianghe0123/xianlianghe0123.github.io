func movingCount(m int, n int, k int) int {
	visit := make(map[[2]int]struct{})
	var dfs func(i, j int) int
	dfs = func(i, j int) int {
		// Ë∂äÁïå
		if i >= m || j >= n {
			return 0
		}
		// Ââ™Êûù
		if _, ok := visit[[2]int{i, j}]; ok {
			return 0
		}
		sum := 0
		for t1, t2 := i, j; t1 > 0 || t2 > 0; t1, t2 = t1/10, t2/10 {
			sum += t1%10 + t2%10
		}
		if sum > k {
			return 0
		}
        // ÁªßÁª≠ÈÅçÂéÜ
		visit[[2]int{i, j}] = struct{}{}
		return 1 + dfs(i+1, j) + dfs(i, j+1)
	}
	return dfs(0, 0)
}
