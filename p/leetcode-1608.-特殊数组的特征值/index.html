func specialArray(nums []int) int {
	sort.Sort(sort.Reverse(sort.IntSlice(nums)))
	for i := range nums {
		if nums[i] >= i+1 && (i+1 == len(nums) || nums[i+1] <= i) {
			return i + 1
		}
	}
	return -1
}
