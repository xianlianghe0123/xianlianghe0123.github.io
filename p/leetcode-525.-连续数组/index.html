func findMaxLength(nums []int) int {
	pos := map[int]int{
		0: -1,
	}
	sum := 0
	max := 0
	for i, num := range nums {
		if num == 0 {
			sum--
		} else {
			sum++
		}
		if p, ok := pos[sum]; ok {
			if max < (i - p) {
				max = i - p
			}
		} else {
			pos[sum] = i
		}
	}
	return max
}
