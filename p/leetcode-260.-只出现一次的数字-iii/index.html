func singleNumber(nums []int) []int {
	cnt := make(map[int]int)
	for _, num := range nums {
		cnt[num]++
	}
	ans := make([]int, 0, 2)
	for num, c := range cnt {
		if c == 1 {
			ans = append(ans, num)
		}
	}
	return ans
}
