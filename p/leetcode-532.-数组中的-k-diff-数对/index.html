func findPairs(nums []int, k int) int {
	ans := make(map[int]struct{})
	exist := make(map[int]struct{})
	for _, num := range nums {
		if _, ok := exist[num-k]; ok {
			ans[num] = struct{}{}
		}
		if _, ok := exist[num+k]; ok {
			ans[num+k] = struct{}{}
		}
		exist[num] = struct{}{}
	}
	return len(ans)
}
