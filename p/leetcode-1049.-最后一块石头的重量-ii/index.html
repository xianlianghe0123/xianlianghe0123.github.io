func lastStoneWeightII(stones []int) int {
	sum := 0
	for _, stone := range stones {
		sum += stone
	}
	dp := make([]int, sum/2+1)
	for _, stones := range stones {
		for j := len(dp) - 1; j >= 0; j-- {
			if j >= stones && dp[j] < dp[j-stones]+stones {
				dp[j] = dp[j-stones] + stones
			}
		}
	}
	if (dp[len(dp)-1] << 1) > sum {
		return (dp[len(dp)-1] << 1) - sum
	} else {
		return sum - (dp[len(dp)-1] << 1)
	}
}
