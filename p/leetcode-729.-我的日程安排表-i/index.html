type MyCalendar struct {
	book [][2]int
}

func Constructor() MyCalendar {
	return MyCalendar{
		book: make([][2]int, 0, 10),
	}
}

func (this *MyCalendar) Book(start int, end int) bool {
	for _, schedule := range this.book {
		if start < schedule[1] && end > schedule[0] {
			return false
		}
	}
	this.book = append(this.book, [2]int{start, end})
	return true
}
