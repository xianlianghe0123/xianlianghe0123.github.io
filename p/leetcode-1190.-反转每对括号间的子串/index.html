func reverseParentheses(s string) string {
	stack := make([]byte, 0, len(s))
	queue := make([]byte, 0, len(s))
	for i := range s {
		if s[i] != ')' {
			stack = append(stack, s[i])
			continue
		}
		queue = queue[:0]
		for j := len(stack) - 1; stack[j] != '('; j-- {
			queue = append(queue, stack[j])
			stack = stack[:len(stack)-1]
		}
		stack = stack[:len(stack)-1]
		stack = append(stack, queue...)
	}
	return string(stack)
}

func reverseParentheses(s string) string {
	stack := make([]int, 0, len(s))
	pair := make(map[int]int)
	for i := range s {
		switch s[i] {
		case '(':
			stack = append(stack, i)
		case ')':
			pair[i] = stack[len(stack)-1]
			pair[stack[len(stack)-1]] = i
			stack = stack[:len(stack)-1]
		}
	}
	res := make([]byte, 0, len(s))
	step := 1
	for i := 0; i < len(s); i += step {
		if s[i] == '(' || s[i] == ')' {
			step = -step
			i = pair[i]
		} else {
			res = append(res, s[i])
		}
	}
	return string(res)
}
