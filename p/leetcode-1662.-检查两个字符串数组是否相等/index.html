func arrayStringsAreEqual(word1 []string, word2 []string) bool {
	for i, j := 0, 0; len(word1) > 0 && len(word2) > 0; {
		if word1[0][i] != word2[0][j] {
			return false
		}
		i++
		if i >= len(word1[0]) {
			word1 = word1[1:]
			i = 0
		}
		j++
		if j >= len(word2[0]) {
			word2 = word2[1:]
			j = 0
		}
	}
	if len(word1) != 0 || len(word2) != 0 {
		return false
	}
	return true
}
