func ambiguousCoordinates(s string) []string {
	s = s[1 : len(s)-1]
	ans := make([]string, 0, len(s)-2)
	for i := len(s) - 1; i > 0; i-- {
		x := s[:i]
		y := s[i:]
		proc(x, func(xx string) {
			proc(y, func(yy string) {
				ans = append(ans, fmt.Sprintf("(%s, %s)", xx, yy))
			})
		})
	}
	return ans
}

func proc(s string, f func(s string)) {
	for j := len(s) - 1; j >= 0; j-- {
		ss := s
		if j > 0 {
			ss = fmt.Sprintf("%s.%s", s[:j], s[j:])
		}
		if len(ss) > 1 {
			if ss[0] == '0' && ss[1] != '.' {
				continue
			}
			if strings.HasSuffix(ss, "0") && strings.IndexByte(ss, '.') != -1 {
				continue
			}
		}
		f(ss)
	}
}
