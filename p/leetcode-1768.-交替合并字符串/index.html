func mergeAlternately(word1 string, word2 string) string {
	minLength := len(word1)
	if minLength > len(word2) {
		minLength = len(word2)
	}
	ans := make([]byte, 0, len(word1)+len(word2))
	for i := 0; i < minLength; i++ {
		ans = append(ans, word1[i], word2[i])
	}
	ans = append(ans, word1[minLength:]...)
	ans = append(ans, word2[minLength:]...)
	return string(ans)
}
