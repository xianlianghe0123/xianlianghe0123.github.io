func findMaximumXOR(nums []int) int {
	max := 0
	for i := range nums {
		for j := i + 1; j < len(nums); j++ {
			t := nums[i] ^ nums[j]
			if max < t {
				max = t
			}
		}
	}
	return max
}

func findMaximumXOR(nums []int) int {
	type Node struct {
		Left, Right *Node
	}
	// ÁîüÊàêTrieÊï∞
	root := new(Node)
	for _, num := range nums {
		node := root
		for bit := 1 << 30; bit > 0; bit >>= 1 {
			if num&bit == 0 {
				if node.Left == nil {
					node.Left = new(Node)
				}
				node = node.Left
			} else {
				if node.Right == nil {
					node.Right = new(Node)
				}
				node = node.Right
			}
		}
	}
	// Ê±ÇÊúÄÂ§ßÂÄº
	max := 0
	for _, num := range nums {
		node := root
		localMax := 0
		for bit := 1 << 30; bit > 0; bit >>= 1 {
			if num&bit == 0 {
				if node.Right != nil {
					node = node.Right
					localMax += bit
				} else {
					node = node.Left
				}
			} else {
				if node.Left != nil {
					node = node.Left
					localMax += bit
				} else {
					node = node.Right
				}
			}
		}
		if max < localMax {
			max = localMax
		}
	}
	return max
}
