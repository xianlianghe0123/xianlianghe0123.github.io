func jobScheduling(startTime []int, endTime []int, profit []int) int {
	jobs := make([][]int, 0, len(startTime))
	for i := range startTime {
		jobs = append(jobs, []int{startTime[i], endTime[i], profit[i]})
	}
	sort.Slice(jobs, func(i, j int) bool {
		return jobs[i][1] < jobs[j][1]
	})
	dp := make([]int, len(startTime)+1)
	for i := range jobs {
		// ‰∫åÂàÜÊü•ÊâæÔºåÂΩìÂâçÂ∑•‰ΩúÂºÄÂßãÊó∂Èó¥‰πãÂâçÊúÄÊôöÁªìÊùüÁöÑÂ∑•‰Ωú
		k := sort.Search(i, func(j int) bool {
			return jobs[j][1] > jobs[i][0]
		})

		dp[i+1] = dp[i]
		if t := dp[k] + jobs[i][2]; dp[i+1] < t {
			dp[i+1] = t
		}
	}
	return dp[len(startTime)]
}
