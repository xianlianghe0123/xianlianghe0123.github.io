func kthLargestValue(matrix [][]int, k int) int {
	vals := make([]int, 0, len(matrix)*len(matrix[0]))
	line := make([]int, len(matrix[0]))
	for a := 0; a < len(matrix); a++ {
		xor := 0
		for b := 0; b < len(matrix[0]); b++ {
			xor ^= matrix[a][b]
			line[b] ^= xor
			vals = append(vals, line[b])
		}
	}
	sort.Ints(vals)
	return vals[len(vals)-k]
}
