func knightProbability(n int, k int, row int, column int) float64 {
	dirs := [][]int{{-1, -2}, {-2, -1}, {-2, 1}, {-1, 2}, {1, 2}, {2, 1}, {2, -1}, {1, -2}}
	dp := make([][]float64, n)
	dp[row] = make([]float64, n)
	dp[row][column] = 1
	for ; k > 0; k-- {
		dpTmp := make([][]float64, n)
		for i := range dp {
			for j := range dp[i] {
				// ‰∏ä‰∏ÄÊ≠•‰∏çÂèØÂà∞ËææÊ¨°ÂçïÂÖÉÊ†º
				if dp[i][j] == 0 {
					continue
				}
				for _, d := range dirs {
					next := []int{i + d[0], j + d[1]}
					// Ë∂äÁïåÂà§Êñ≠
					if next[0] < 0 || next[0] >= n || next[1] < 0 || next[1] >= n {
						continue
					}
					// ËΩ¨ÁßªÂà∞‰∏ã‰∏™ÂçïÂÖÉÊ†ºÁöÑÊ¶ÇÁéáÁ¥ØÂä†
					if len(dpTmp[next[0]]) == 0 {
						dpTmp[next[0]] = make([]float64, n)
					}
					dpTmp[next[0]][next[1]] += dp[i][j] * 0.125
				}
			}
		}
		dp = dpTmp
	}
	// Ê¶ÇÁéáÊ±ÇÂíå
	sum := 0.
	for i := range dp {
		for j := range dp[i] {
			sum += dp[i][j]
		}
	}
	return sum
}
