func loudAndRich(richer [][]int, quiet []int) []int {
	graph := make([][]int, len(quiet))
	inDegrees := make(map[int]int)
	ans := make([]int, len(quiet))
	for i := range quiet {

	for len(inDegrees) > 0 {

			}
			break
		}
	}
	return ans
}
