func displayTable(orders [][]string) [][]string {
	tableFoodsCnt := make(map[int]map[string]int)
	foods := make([]string, 0, len(orders)) // ËèúÂìÅÂêçÂçï
	isAddToFoods := make(map[string]bool)   // ÂéªÈáç
	for _, order := range orders {
		// Âä†ÂÖ•ËèúÂìÅÂêçÂçï
		if !isAddToFoods[order[2]] {
			foods = append(foods, order[2])
			isAddToFoods[order[2]] = true
		}
		// ÁªüËÆ°Êï∞Èáè
		table, _ := strconv.Atoi(order[1])
		if tableFoodsCnt[table] == nil {
			tableFoodsCnt[table] = make(map[string]int)
		}
		tableFoodsCnt[table][order[2]]++
	}
	// ËèúÂìÅÊéíÂ∫è
	sort.Strings(foods)
	// Ê°åÂè∑ÊéíÂ∫è
	tables := make([]int, 0, len(tableFoodsCnt))
	for table := range tableFoodsCnt {
		tables = append(tables, table)
	}
	sort.Ints(tables)
	// ÁîüÊàêÁªìÊûú
	res := make([][]string, len(tables)+1)
	res[0] = make([]string, 0, len(foods)+1)
	// Ë°®Â§¥
	res[0] = append(res[0], "Table")
	for _, food := range foods {
		res[0] = append(res[0], food)
	}
	// Â°´Ë°®
	i := 1
	for _, table := range tables {
		res[i] = make([]string, 0, len(foods)+1)
		res[i] = append(res[i], strconv.FormatInt(int64(table), 10)) // Ê°åÂè∑
		// ËèúÂìÅÊï∞
		for _, food := range foods {
			res[i] = append(res[i], strconv.FormatInt(int64(tableFoodsCnt[table][food]), 10))
		}
		i++
	}
	return res
}
