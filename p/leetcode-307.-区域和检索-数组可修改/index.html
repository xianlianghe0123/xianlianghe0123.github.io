type NumArray struct {
	nums        []int
	segmentSize int
	segmentSum  []int
}

func Constructor(nums []int) NumArray {
	numArray := NumArray{
		nums: nums,
	}
	numArray.segmentSize = int(math.Sqrt(float64(len(nums))))
	numArray.segmentSum = make([]int, 0, len(nums)/numArray.segmentSize+1)
	// åˆ†æ®µæ±‚å’Œ
	for i := 0; i < len(nums); i += numArray.segmentSize {
		segmentSum := 0
		j := i + numArray.segmentSize - 1
		// æœ€åä¸€æ®µä¸è¶³
		if j >= len(nums) {
			j = len(nums) - 1

		for ; j >= i; j-- {
			segmentSum += nums[j]
		}
		numArray.segmentSum = append(numArray.segmentSum, segmentSum)
	}
	return numArray
}

func (this *NumArray) Update(index int, val int) {
	this.segmentSum[index/this.segmentSize] += val - this.nums[index] // æ›´æ–°è¯¥æ®µå’Œ

}
