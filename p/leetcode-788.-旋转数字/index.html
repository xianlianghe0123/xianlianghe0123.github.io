func valid(n int) bool {
	same := true
	for ; n > 0; n /= 10 {
		switch n % 10 {
		case 3, 4, 7:
			return false
		case 2, 5, 6, 9:
			same = false
		}
	}
	return !same
}

func rotatedDigits(n int) int {
	ans := 0
	for i := 1; i <= n; i++ {
		if valid(i) {
			ans++
		}
	}
	return ans
}
