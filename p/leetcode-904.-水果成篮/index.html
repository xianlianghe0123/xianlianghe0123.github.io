func totalFruit(fruits []int) int {
	cnt := make(map[int]int)
	ans := 0
	for l, r := 0, 0; r < len(fruits); r++ {
		cnt[fruits[r]]++
		// Êî∂Áº©Â∑¶Á´ØÁÇπ
		for ; len(cnt) > 2; l++ {
			cnt[fruits[l]]--
			if cnt[fruits[l]] == 0 {
				delete(cnt, fruits[l])
			}
		}
		if ans < r-l+1 {
			ans = r - l + 1
		}
	}
	return ans
}
