func dominantIndex(nums []int) int {
	idxOfMax := 0
	for i, num := range nums {
		if nums[idxOfMax] < num {
			idxOfMax = i
		}
	}
	for i, num := range nums {
		if i != idxOfMax && nums[idxOfMax] < num<<1 {
			return -1
		}
	}
	return idxOfMax
}
