func reverse(x int) int {
	res := 0
	for i := 0; x != 0 && i < 9; i++ {
		res = res*10 + x%10
		x /= 10
	}
	if x != 0 {
		if (x > 0 && (res > math.MaxInt32/10 || (res == math.MaxInt32/10 && x > math.MaxInt32%10))) ||
			(x < 0 && (res < math.MinInt32/10 || (res == math.MinInt32/10 && x < math.MinInt32%10))) {
			res = 0
		} else {
			res = res*10 + x
		}
	}
	return res
}
