func goodDaysToRobBank(security []int, time int) []int {
	descDays := make([]int, len(security))
	ascDays := make([]int, len(security))
	for i := 1; i < len(security); i++ {
		if security[i] <= security[i-1] {
			descDays[i] = descDays[i-1] + 1
		}
		if security[len(security)-1-i] <= security[len(security)-i] {
			ascDays[len(security)-1-i] = ascDays[len(security)-i] + 1
		}
	}
	ans := make([]int, 0, len(security))
	for i := len(security) - 1 - time; i >= time; i-- {
		if descDays[i] >= time && ascDays[i] >= time {
			ans = append(ans, i)
		}
	}
	return ans
}
