func myPow(x float64, n int) float64 {
	ans := float64(1)
	pow := float64(x)
	if n < 0 {
		pow = 1 / pow
		n = -n
	}
	for ; n != 0; n >>= 1 {
		if n&1 == 1 {
			ans *= pow
		}
		pow *= pow
	}
	return ans
}
