func maxPathSum(root *TreeNode) int {
	var dfs func(node *TreeNode, prefixSum int) (maxPath, maxPrefixSum int)
	dfs = func(node *TreeNode, prefixSum int) (maxPath, maxPrefixSum int) {
		if node == nil {
			return -math.MaxInt32, prefixSum
		}
        // å­æ ‘çš„æœ€å¤§å‰ç¼€å’Œã€æœ€å¤§è·¯å¾„å’Œ
		maxPath, maxPrefixSum = dfs(node.Left, prefixSum+node.Val)
		rMaxPath, rMaxPrefixSum := dfs(node.Right, prefixSum+node.Val)
        // å½“å‰æ ‘çš„å¤§è·¯å¾„å’Œ
		curMaxPath := maxPrefixSum + rMaxPrefixSum - (prefixSum << 1) - node.Val
		if maxPath < curMaxPath {
			maxPath = curMaxPath
		}
		if maxPath < rMaxPath {
			maxPath = rMaxPath
		}
        // æœ€å¤§å‰ç¼€å’Œ
		if maxPrefixSum < rMaxPrefixSum {
			maxPrefixSum = rMaxPrefixSum
		}
		if maxPrefixSum < prefixSum {
			maxPrefixSum = prefixSum
		}
		return
	}
	maxPath, _ := dfs(root, 0)
	return maxPath
}
