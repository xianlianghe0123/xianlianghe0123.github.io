func isAlienSorted(words []string, order string) bool {
	weight := make(map[byte]int)
	for i := range order {
		weight[order[i]] = i
	}
	cmp := func(s1 string, s2 string) int {
		i := 0
		for ; i < len(s1) && i < len(s2); i++ {
			if weight[s1[i]] < weight[s2[i]] {
				return -1
			} else if weight[s1[i]] > weight[s2[i]] {
				return 1
			}
		}
		if len(s1) == len(s2) {
			return 0
		} else if len(s1) < len(s2) {
			return -1
		} else {
			return 1
		}
	}
	for i := 1; i < len(words); i++ {
		if cmp(words[i-1], words[i]) > 0 {
			return false
		}
	}
	return true
}
