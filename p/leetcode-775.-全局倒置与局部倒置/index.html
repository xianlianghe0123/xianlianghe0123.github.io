func isIdealPermutation(nums []int) bool {
	minIdx := len(nums) - 1
	for i := len(nums) - 2; i >= 0; i-- {
		if nums[i] > nums[minIdx] {
			if minIdx-i > 1 {
				return false
			}
		}
		if nums[i+1] < nums[minIdx] {
			minIdx = i + 1
			continue
		}
	}
	return true
}
