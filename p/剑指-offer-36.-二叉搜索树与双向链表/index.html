class Solution {
public:
    Node *treeToDoublyList(Node *root) {
        if (root == nullptr) {
            return nullptr;
        }
        dfs(root);
        head->left = tail;
        tail->right = head;
        return head;
    }

private:
    Node *head, *tail;
    void dfs(Node *node) {
        if(node == nullptr) {
            return;
        }
        dfs(node->left);
        if (head == nullptr) {
            head = node;
        }
        if (tail == nullptr) {
            tail = node;
        } else {
            tail->right = node;
            node->left = tail;
            tail = node;
        }
        dfs(node->right);
    }
};
