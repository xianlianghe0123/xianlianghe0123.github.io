func exclusiveTime(n int, logs []string) []int {
	ans := make([]int, n)
	stack := make([]int, 0, len(logs)/2)
	now := 0 // ÂΩìÂâçÊó∂Èó¥ÔºåÂºÄÂßã‰ΩçÁΩÆ
	for _, log := range logs {
		l := strings.Split(log, ":")
		no, _ := strconv.Atoi(l[0])
		time, _ := strconv.Atoi(l[2])
		// ÂΩìÂâçÂú®ÊâßË°åÁöÑÂáΩÊï∞ÔºåÊâßË°åÊó∂Èó¥Á¥ØÂä†
		if len(stack) > 0 {
			cur := stack[len(stack)-1]
			ans[cur] += time - now
			if l[1] == "end" {
				ans[cur]++
			}
		}

		if l[1] == "start" {
			// Êñ∞ÂáΩÊï∞ËøêË°åÂºÄÂßãÔºåÂÖ•Ê†à
			stack = append(stack, no)
			now = time
		} else {
			// Ê†àÈ°∂ÂáΩÊï∞ËøêË°åÁªìÊùüÔºåÂá∫Ê†à
			stack = stack[:len(stack)-1]
			now = time + 1
		}
	}
	return ans
}
