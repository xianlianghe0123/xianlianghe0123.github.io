func finalPrices(prices []int) []int {
	ans := make([]int, len(prices))
	for i := range prices {
		discount := 0
		for j := i + 1; j < len(prices); j++ {
			if prices[j] <= prices[i] {
				discount = prices[j]
				break
			}
		}
		ans[i] = prices[i] - discount
	}
	return ans
}
