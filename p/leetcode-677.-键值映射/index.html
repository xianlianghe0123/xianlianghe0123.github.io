type MapSum struct {
	hash map[string]int
}

func Constructor() MapSum {
	return MapSum{
		hash: make(map[string]int),
	}
}

func (this *MapSum) Insert(key string, val int) {
	this.hash[key] = val
}

func (this *MapSum) Sum(prefix string) int {
	sum := 0
	for key, val := range this.hash {
		if strings.HasPrefix(key, prefix) {
			sum += val
		}
	}
	return sum
}

type Node struct {
	val   int
	child map[byte]*Node
}

type MapSum struct {
	head *Node
}

func Constructor() MapSum {
	return MapSum{
		head: &Node{
			child: make(map[byte]*Node),
		},
	}
}

func (this *MapSum) Insert(key string, val int) {
	node := this.head
	for i := 0; i < len(key); i, node = i+1, node.child[key[i]] {
		if _, ok := node.child[key[i]]; !ok {
			node.child[key[i]] = &Node{
				val:   0,
				child: make(map[byte]*Node),
			}
		}
	}
	node.val = val
}

func (this *MapSum) Sum(prefix string) int {
	node := this.head
	for i := 0; i < len(prefix); i++ {
		var ok bool
		if node, ok = node.child[prefix[i]]; !ok {
			return 0
		}
	}
	var dfs func(node *Node) int
	dfs = func(node *Node) int {
		sum := node.val
		for _, node := range node.child {
			sum += dfs(node)
		}
		return sum
	}
	return dfs(node)
}
