func customSortString(order string, s string) string {
	w := make([]int, 26)
	for i := range order {
		w[order[i]-'a'] = i
	}
	t := []byte(s)
	sort.Slice(t, func(i, j int) bool {
		return w[t[i]-'a'] < w[t[j]-'a']
	})
	return string(t)
}

func customSortString(order string, s string) string {
	cnt := make([]int, 26)
	for i := range s {
		cnt[s[i]-'a']++
	}
	ans := make([]byte, 0, len(s))
	for i := range order {
		for ; cnt[order[i]-'a'] > 0; cnt[order[i]-'a']-- {
			ans = append(ans, order[i])
		}
	}
	for i := range cnt {
		for ; cnt[i] > 0; cnt[i]-- {
			ans = append(ans, byte(i)+'a')
		}
	}
	return string(ans)
}
