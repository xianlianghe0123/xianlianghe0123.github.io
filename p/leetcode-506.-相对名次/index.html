func findRelativeRanks(score []int) []string {
	t := make([]int, 0, len(score))
	for _, s := range score {
		t = append(t, s)
	}
	copy(t, score)
	sort.Ints(t)
	rank := make(map[int]int)
	for i := len(t) - 1; i >= 0; i-- {
		rank[t[i]] = len(t) - i
	}
	ans := make([]string, 0, len(score))
	for _, s := range score {
		r := ""
		switch rank[s] {
		case 1:
			r = "Gold Medal"
		case 2:
			r = "Silver Medal"
		case 3:
			r = "Bronze Medal"
		default:
			r = strconv.FormatInt(int64(rank[s]), 10)
		}
		ans = append(ans, r)
	}
	return ans
}
