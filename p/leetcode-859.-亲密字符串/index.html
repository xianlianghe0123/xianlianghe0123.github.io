func buddyStrings(s string, goal string) bool {
	if len(s) != len(goal) {
		return false
	}
	diff := make([]int, 0, 2)
	repeat := false
	exist := make(map[byte]struct{})
	for i := range s {
		if s[i] == goal[i] {
			if !repeat {
				if _, ok := exist[s[i]]; ok {
					repeat = true
				}
				exist[s[i]] = struct{}{}
			}
			continue
		}
		if len(diff) == 2 {
			return false
		}
		diff = append(diff, i)
	}
	switch len(diff) {
	case 0:
		return repeat
	case 2:
		return s[diff[0]] == goal[diff[1]] && s[diff[1]] == goal[diff[0]]
	}
	return false
}
