func countDigitOne(n int) int {
	ans := 0
	for i, x := 1, n; x > 0; x, i = x/10, i*10 {
		ans += x / 10 * i
		switch x % 10 {
		case 0:
		case 1:
			ans += n%i + 1
		default:
			ans += i
		}
	}
	return ans
}
