	}

func (this *CBTInserter) Insert(val int) int {
	q := list.New()
	q.PushBack(this.root)
	for q.Len() > 0 {
		n := q.Remove(q.Front()).(*TreeNode)
		if n.Left == nil {
			n.Left = &TreeNode{
				Val: val,
			}
			return n.Val
		}
		if n.Right == nil {
			n.Right = &TreeNode{
				Val: val,
			}
			return n.Val
		}
		q.PushBack(n.Left)
		q.PushBack(n.Right)
	}
	return 0
}

func (this *CBTInserter) Get_root() *TreeNode {
	return this.root
}

type CBTInserter struct {
	nodes []*TreeNode
}

func Constructor(root *TreeNode) CBTInserter {
	nodes := make([]*TreeNode, 0, 1)
	q := list.New()
	q.PushBack(root)
	for q.Len() > 0 {
		n := q.Remove(q.Front()).(*TreeNode)
		nodes = append(nodes, n)
		if n.Left != nil {
			q.PushBack(n.Left)
		}
		if n.Right != nil {
			q.PushBack(n.Right)
		}
	}
	return CBTInserter{
		nodes: nodes,
	}
}

func (this *CBTInserter) Insert(val int) int {
	n := &TreeNode{
		Val: val,
	}
	this.nodes = append(this.nodes, n)
	parent := this.nodes[(len(this.nodes)-2)/2]
	if parent.Left == nil {
		parent.Left = n
	} else {
		parent.Right = n
	}
	return parent.Val
}

func (this *CBTInserter) Get_root() *TreeNode {
	return this.nodes[0]
}
