func validIPAddress(queryIP string) string {
	if strings.Contains(queryIP, ".") { // Âà§ÂÆöÊòØÂê¶‰∏∫IPV4
		items := strings.Split(queryIP, ".")
		if len(items) != 4 {
			return "Neither"
		}
		for _, item := range items {
			// ‰∏çÊòØÊúâÊïàÊï¥Êï∞
			t, err := strconv.Atoi(item)
			if err != nil {
				return "Neither"
			}
			if t < 0 || t > 255 {
				return "Neither"
			}
			// ÂåÖÂê´ÂâçÂØº0
			if item[0] == '0' && len(item) > 1 {
				return "Neither"
			}
		}
		return "IPv4"
	} else if strings.Contains(queryIP, ":") { // Âà§Êñ≠ÊòØÂê¶‰∏∫ IPV6
		items := strings.Split(queryIP, ":")
		if len(items) != 8 {
			return "Neither"
		}
		for _, item := range items {
			// ‰∏çÊòØÊúâÊïàÈïøÂ∫¶
			if len(item) == 0 || len(item) > 4 {
				return "Neither"
			}
			// ÂåÖÂê´Êó†ÊïàÂ≠óÁ¨¶
			for i := range item {
				if !((item[i] >= '0' && item[i] <= '9') || (item[i] >= 'a' && item[i] <= 'f') || (item[i] >= 'A' && item[i] <= 'F')) {
					return "Neither"
				}
			}
		}
		return "IPv6"
	}
	return "Neither"
}
