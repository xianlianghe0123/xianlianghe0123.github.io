func checkSubarraySum(nums []int, k int) bool {
	pos := map[int]int{
		0: -1,
	}
	sum := 0
	for i := 0; i < len(nums); i++ {
		sum = (sum + nums[i]) % k
		if p, ok := pos[sum]; ok {
			if i-p > 1 {
				return true
			}
		} else {
			pos[sum] = i
		}
	}
	return false
}
