func minMutation(start string, end string, bank []string) int {
	chars := []byte("ACGT")
	valid := make(map[string]struct{})
	for _, b := range bank {
		valid[b] = struct{}{}
	}
	q := list.New()
	q.PushBack(start)
	visit := make(map[string]struct{})
	visit[start] = struct{}{}
	for step := 0; q.Len() > 0; step++ {
		for i := q.Len(); i > 0; i-- {
			s := q.Remove(q.Front()).(string)
			if s == end {
				return step
			}
			// Â∞ùËØïÂèòÊç¢ÊØè‰∏Ä‰∏™Â≠óÁ¨¶
			for idx := range s {
				t := []byte(s)
				// Â∞ÜÂØπÂ∫îÂ≠óÁ¨¶ÂèòÊç¢‰∏∫‰ªªÊÑè‰∏Ä‰∏™ÂÖ∂‰ªñÂ≠óÁ¨¶
				for _, char := range chars {
					if t[idx] == char {
						continue
					}
					t[idx] = char
					next := string(t)
					// Â∑≤ÊêúÁ¥¢Ëøá
					if _, ok := visit[next]; ok {
						continue
					}
					visit[next] = struct{}{}
					// Êó†ÊïàÂ∫èÂàó
					if _, ok := valid[next]; !ok {
						continue
					}
					q.PushBack(next)
				}
			}
		}
	}
	return -1
}
