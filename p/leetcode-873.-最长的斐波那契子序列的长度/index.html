func lenLongestFibSubseq(arr []int) int {
	exist := make(map[int]struct{})
	for _, num := range arr {
		exist[num] = struct{}{}
	}

	ans := 0
	for i := len(arr) - 3; i >= 0; i-- {
		for j := len(arr) - 2; j > i; j-- {
			a, b := arr[i], arr[j]
			length := 2
			for {
				if _, ok := exist[a+b]; !ok {
					break
				}
				length++
				a, b = b, a+b
			}
			if length >= 3 && ans < length {
				ans = length
			}
		}
	}
	return ans
}
