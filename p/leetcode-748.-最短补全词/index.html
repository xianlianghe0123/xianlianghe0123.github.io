func shortestCompletingWord(licensePlate string, words []string) string {
	charCnt := make(map[byte]int)
	for i := range licensePlate {
		if 'A' <= licensePlate[i] && licensePlate[i] <= 'Z' {
			charCnt[licensePlate[i]-'A'+'a']++
		} else if 'a' <= licensePlate[i] && licensePlate[i] <= 'z' {
			charCnt[licensePlate[i]]++
		}
	}
	ans := ""
Outer:
	for _, word := range words {
		t := make(map[byte]int)
		for i := range word {
			t[word[i]]++
		}
		for k, v := range charCnt {
			if t[k] < v {
				continue Outer
			}
		}
		if ans == "" || len(ans) > len(word) {
			ans = word
		}
	}
	return ans
}
