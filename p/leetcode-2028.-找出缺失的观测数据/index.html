func missingRolls(rolls []int, mean int, n int) []int {
	sum := 0
	for _, r := range rolls {
		sum += r
	}
	sum = mean*(n+len(rolls)) - sum
	// Ê≤°ÊúâÁ≠îÊ°à
	if sum < n || sum > 6*n {
		return nil
	}
	point := sum / n // Âπ≥ÂùáÁÇπÊï∞
	more := sum % n  // Â§ö‰ΩôÁÇπÊï∞
	ans := make([]int, n)
	for i := range ans {
		ans[i] = point
		// Â§ö‰ΩôÁÇπÊï∞ÂùáÂåÄÂàÜÈÖç
		if more > 0 {
			ans[i]++
			more--
		}
	}
	return ans
}
