func modifyString(s string) string {
	t := []byte(s)
	for i := 0; i < len(t); i++ {
		if t[i] != '?' {
			continue
		}
		// Ââç‰∏Ä‰∏™Â≠óÁ¨¶
		prev := byte('$')
		if i > 0 {

		// Âêé‰∏Ä‰∏™Â≠óÁ¨¶
		post := byte('$')
		if i < len(t)-1 && t[i+1] != '?' {
			post = t[i+1]
		}
		// ÂØªÊâæÂèØ‰ª•ÊõøÊç¢ÁöÑÂ≠óÁ¨¶

		}
	}
	return string(t)
}
