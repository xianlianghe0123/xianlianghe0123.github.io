func CheckPermutation(s1 string, s2 string) bool {
	if len(s1) != len(s2) {
		return false
	}
	cnt := make(map[byte]int)
	for i := range s1 {
		cnt[s1[i]]++
	}
	for i := range s2 {
		cnt[s2[i]]--
		if cnt[s2[i]] < 0 {
			return false
		}
		if cnt[s2[i]] == 0 {
			delete(cnt, s2[i])
		}
	}
	return len(cnt) == 0
}
