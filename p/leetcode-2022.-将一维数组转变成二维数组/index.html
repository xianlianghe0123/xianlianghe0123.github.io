func construct2DArray(original []int, m int, n int) [][]int {
	if m*n != len(original) {
		return make([][]int, 0)
	}
	matrix := make([][]int, 0, m)
	for i := 0; i < len(original); i += n {
		matrix = append(matrix, original[i:i+n])
	}
	return matrix
}
