func circularArrayLoop(nums []int) bool {
	visit := make([]bool, len(nums))
	for i := 0; i < len(nums); i++ {
		if visit[i] {
			continue
		}
		slow := i
		visit[slow] = true
		fast := ((slow+nums[slow])%len(nums) + len(nums)) % len(nums)
		if nums[slow]*nums[fast] < 0 || visit[fast] {
			continue
		}
		for slow != fast {
			// Âø´ÊåáÈíàÂâçËøõ‰∏§Ê≠•
			next := ((fast+nums[fast])%len(nums) + len(nums)) % len(nums)
			if next == fast || nums[fast]*nums[next] < 0 {
				break
			}
			fast = next
			next = ((fast+nums[fast])%len(nums) + len(nums)) % len(nums)
			if next == fast || nums[fast]*nums[next] < 0 {
				break
			}
			fast = next
			// ÊÖ¢ÊåáÈíàÂâçËøõ‰∏ÄÊ≠•
			slow = ((slow+nums[slow])%len(nums) + len(nums)) % len(nums)
			if visit[slow] {
				break
			}
			visit[slow] = true
		}
		if slow == fast {
			return true
		}
	}
	return false
}
