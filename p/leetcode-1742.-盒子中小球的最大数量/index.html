func countBalls(lowLimit int, highLimit int) int {
	cnt := make(map[int]int)
	ans := 0
	for i := lowLimit; i <= highLimit; i++ {
		digitSum := 0
		for x := i; x > 0; x /= 10 {
			digitSum += x % 10
		}
		cnt[digitSum]++
		if ans < cnt[digitSum] {
			ans = cnt[digitSum]
		}
	}
	return ans
}
