	m, n := len(heights), len(heights[0])
	visit := make([][]int, m)
	for i := range heights {
		visit[i] = make([]int, n)
	}
	dirs := [][]int{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}
	var dfs func(row, col int, flag int)
	dfs = func(row, col int, flag int) {
		visit[row][col] |= flag
		// Ë∂äÁïåÊ£ÄÊü•

		for _, dir := range dirs {
			next := []int{row + dir[0], col + dir[1]}
