func findDuplicateSubtrees(root *TreeNode) []*TreeNode {
	ans := make([]*TreeNode, 0, 1)
	exist := make(map[string]int)
	var dfs func(node *TreeNode) string
	dfs = func(node *TreeNode) string {
		if node == nil {
			return ""
		}
		s := fmt.Sprintf("(%s)%d(%s)", dfs(node.Left), node.Val, dfs(node.Right))
		exist[s]++
		if exist[s] == 2 {
			ans = append(ans, node)
		}
		return s
	}
	dfs(root)
	return ans
}
