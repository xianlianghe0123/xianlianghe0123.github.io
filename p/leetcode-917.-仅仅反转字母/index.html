func reverseOnlyLetters(s string) string {
	t := []byte(s)
	for i, j := 0, len(t)-1; i < j; {
		if !unicode.IsLetter(rune(t[i])) {
			i++
			continue
		}
		if !unicode.IsLetter(rune(t[j])) {
			j--
			continue
		}
		t[i], t[j] = t[j], t[i]
		i++
		j--
	}
	return string(t)
}
