func maximizeXor(nums []int, queries [][]int) []int {
	type Node struct {
		Left, Right *Node
		Min         int
	}
	// ç”ŸæˆTrieæ•°
	root := new(Node)
	for _, num := range nums {
		node := root
		for bit := 1 << 30; bit > 0; bit >>= 1 {
			if num&bit == 0 {
				if node.Left == nil {
					node.Left = &Node{
						Left:  nil,
						Right: nil,
						Min:   num,
					}
				}
				node = node.Left
				if node.Min > num {
					node.Min = num
				}
			} else {
				if node.Right == nil {
					node.Right = &Node{
						Left:  nil,
						Right: nil,
						Min:   num,
					}
				}
				node = node.Right
				if node.Min > num {
					node.Min = num
				}
			}
		}
	}
	res := make([]int, 0, len(queries))
	// æ±‚æœ€å¤§å€¼
	for _, query := range queries {
		node := root
		val := 0
		for bit := 1 << 30; bit > 0; bit >>= 1 {
			if query[0]&bit == 0 { // ä¼˜å…ˆå–1
				if node.Right != nil && node.Right.Min <= query[1] {
					node = node.Right
					val += bit
				} else if node.Left != nil {
					node = node.Left
				} else {
					val = -1
					break
				}
			} else { // ä¼˜å…ˆå–0
				if node.Left != nil {
					node = node.Left
				} else if node.Right != nil && node.Right.Min <= query[1] {
					node = node.Right
					val += bit
				} else {
					val = -1
					break
				}
			}
		}
		if val == -1 {
			res = append(res, -1)
		} else {
			res = append(res, query[0]^val)
		}
	}
	return res
}
