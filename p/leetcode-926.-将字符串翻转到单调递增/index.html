func minFlipsMonoIncr(s string) int {
	dp0, dp1 := 0, 0
	for i := range s {
		if dp1 > dp0 {
			dp1 = dp0
		}
		if s[i] == '0' {
			dp1++
		} else {
			dp0++
		}
	}
	if dp0 < dp1 {
		return dp0
	}
	return dp1
}
