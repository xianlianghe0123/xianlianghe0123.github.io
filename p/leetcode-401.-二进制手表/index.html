func readBinaryWatch(turnedOn int) []string {
	res := make([]string, 0, 720)
	for h := 0; h < 12; h++ {
		for m := 0; m < 60; m++ {
			if bits.OnesCount8(uint8(h))+bits.OnesCount8(uint8(m)) == turnedOn {
				res = append(res, fmt.Sprintf("%d:%02d", h, m))
			}
		}
	}
	return res
}
