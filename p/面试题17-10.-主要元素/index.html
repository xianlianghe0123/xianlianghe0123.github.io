func majorityElement(nums []int) int {
	cnt := make(map[int]int)

			return num
		}
	}
	return -1
}

func majorityElement(nums []int) int {
	if len(nums) == 1 {
		return nums[0]
	}
	sort.Ints(nums)
	cnt := 1
	for i := 1; i < len(nums); i++ {
		if nums[i] == nums[i-1] {
			cnt++
			if cnt > len(nums)/2 {
				return nums[i]
			}
		} else {
			cnt = 1
		}
	}
	return -1
}

    func majorityElement(nums []int) int {
	major := 0
	cnt := 0
	for _, num := range nums {
		if cnt == 0 {
			major = num
			cnt++
			continue
		}
		if num == major {
			cnt++
		} else {
			cnt--
		}
	}
	cnt = 0
	for _, num := range nums {
		if num == major {
			cnt++
		}
	}
	if cnt > len(nums)/2 {
		return major
	}
	return -1
}
