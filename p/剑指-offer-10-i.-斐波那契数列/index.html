func fib(n int) int {
	const MOD = 1e9 + 7
	dp := []int{0, 1, 0}
	if n < 2 {
		return dp[n]
	}
	for i := 2; i <= n; i++ {
		dp[2] = (dp[0] + dp[1]) % MOD
		dp[0], dp[1] = dp[1], dp[2]
	}
	return dp[2]
}
