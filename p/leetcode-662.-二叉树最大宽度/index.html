func widthOfBinaryTree(root *TreeNode) int {
	q := list.New()
	q.PushBack(&TreeNode{
		Val:   1,
		Left:  root.Left,
		Right: root.Right,
	})
	ans := 0
	for q.Len() > 0 {
		min, max := math.MaxInt, math.MinInt
		for i := q.Len(); i > 0; i-- {
			n := q.Remove(q.Front()).(*TreeNode)
			if min > n.Val {
				min = n.Val
			}
			if max < n.Val {
				max = n.Val
			}
			if n.Left != nil {
				q.PushBack(&TreeNode{
					Val:   n.Val * 2,
					Left:  n.Left.Left,
					Right: n.Left.Right,
				})
			}
			if n.Right != nil {
				q.PushBack(&TreeNode{
					Val:   n.Val*2 + 1,
					Left:  n.Right.Left,
					Right: n.Right.Right,
				})
			}
		}
		if ans < max-min+1 {
			ans = max - min + 1
		}
	}
	return ans
}

func widthOfBinaryTree(root *TreeNode) int {
	ans := 1

	minNo := make(map[int]int)
	var dfs func(node *TreeNode, depth int, no int)
	dfs = func(node *TreeNode, depth int, no int) {
		if node == nil {
			return
		}
		if _, ok := minNo[depth]; !ok {
			minNo[depth] = no
		} else if ans < no-minNo[depth]+1 {
			ans = no - minNo[depth] + 1
		}
		dfs(node.Left, depth+1, no*2)
		dfs(node.Right, depth+1, no*2+1)
	}

	dfs(root, 0, 1)
	return ans
}
