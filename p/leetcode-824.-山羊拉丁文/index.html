func toGoatLatin(sentence string) string {
	ans := make([]byte, 0, len(sentence))
	wordCnt := 1
	begin := 0
	for i := 0; i <= len(sentence); i++ {
		if i == len(sentence) || sentence[i] == ' ' {
			switch sentence[begin] {
			case 'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U':
				ans = append(ans, sentence[begin:i]...)
				ans = append(ans, 'm', 'a')
			default:
				ans = append(ans, sentence[begin+1:i]...)
				ans = append(ans, sentence[begin], 'm', 'a')
			}
			for j := 0; j < wordCnt; j++ {
				ans = append(ans, 'a')
			}
			ans = append(ans, ' ')
			begin = i + 1
			wordCnt++
		}
	}
	return string(ans[:len(ans)-1])
}
