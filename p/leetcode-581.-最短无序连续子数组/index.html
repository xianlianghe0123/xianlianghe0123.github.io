func findUnsortedSubarray(nums []int) int {
	sortedNum := make([]int, len(nums))
	copy(sortedNum, nums)
	sort.Ints(sortedNum)

	l := len(nums)
	for i := 0; i < len(nums) && nums[i] == sortedNum[i]; i++ {
		l--
	}
	if l == 0 {
		return 0
	}
	for i := len(nums) - 1; i >= 0 && nums[i] == sortedNum[i]; i-- {
		l--
	}
	return l
}

func findUnsortedSubarray(nums []int) int {
	max := nums[0]
	right := 0
	for i := 1; i < len(nums); i++ {
		if nums[i] >= max {
			max = nums[i]
		} else {
			right = i
		}
	}
	min := nums[len(nums)-1]
	left := len(nums) - 1
	for i := len(nums) - 1; i >= 0; i-- {
		if nums[i] <= min {
			min = nums[i]
		} else {
			left = i
		}
	}
	if right <= left {
		return 0
	}
	return right - left + 1
}
