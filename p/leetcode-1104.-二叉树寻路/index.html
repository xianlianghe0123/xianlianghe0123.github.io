func pathInZigZagTree(label int) []int {
	layer := 0   // Â±Ç
	cnt := layer // Êª°‰∫åÂèâÊ†ëÁªìÁÇπÊï∞Èáè

		layer++
		cnt = 2*(cnt+1) - 1

	idx := layer - 1
	res := make([]int, layer)
	num := label // ÂÆåÂÖ®‰∫åÂèâÊ†ëÁºñÂè∑
	if layer&1 == 0 {

	}
	for layer > 0 {
		if layer&1 == 0 { // ÂÅ∂Êï∞Â±ÇÔºåËΩ¨Êç¢

		layer--
		cnt /= 2
	}
	return res
}
