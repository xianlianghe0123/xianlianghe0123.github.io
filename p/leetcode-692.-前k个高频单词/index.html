func topKFrequent(words []string, k int) []string {
	wordsCnt := make(map[string]int)
	for _, word := range words {
		wordsCnt[word]++
	}
	uniqWords := make([]string, 0, len(wordsCnt))
	for k := range wordsCnt {
		uniqWords = append(uniqWords, k)
	}
	sort.Slice(uniqWords, func(i, j int) bool {
		if wordsCnt[uniqWords[i]] > wordsCnt[uniqWords[j]] {
			return true
		} else if wordsCnt[uniqWords[i]] < wordsCnt[uniqWords[j]] {
			return false
		} else {
			return uniqWords[i] < uniqWords[j]
		}
	})
	return uniqWords[:k]
}
