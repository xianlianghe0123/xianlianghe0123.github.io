func snakesAndLadders(board [][]int) int {
	type Pair struct {
		Pos  int
		Step int
	}
	isVisit := make([]bool, len(board)*len(board[0]))
	queue := make([]*Pair, 0, len(isVisit))
	queue = append(queue, &Pair{
		Pos:  0,
		Step: 0,
	})
	isVisit[0] = true
	for len(queue) > 0 {
		pos := queue[0].Pos
		step := queue[0].Step
		queue = queue[1:]
		if pos == len(isVisit)-1 {
			return step
		}
		for i := 1; i <= 6; i++ {
			if pos+i >= len(isVisit) {
				break
			}
			row := len(board) - 1 - (pos+i)/len(board[0])
			col := (pos + i) % len(board[0])
			if (len(board)-1-row)&1 == 1 {
				col = len(board[0]) - 1 - col
			}
			cell := board[row][col] - 1
			if cell < 0 {
				if isVisit[pos+i] {
					continue
				}
				queue = append(queue, &Pair{
					Pos:  pos + i,
					Step: step + 1,
				})
				isVisit[pos+i] = true
			} else if !isVisit[cell] {
				queue = append(queue, &Pair{
					Pos:  cell,
					Step: step + 1,
				})
				isVisit[cell] = true
			}
		}
	}
	return -1
}
