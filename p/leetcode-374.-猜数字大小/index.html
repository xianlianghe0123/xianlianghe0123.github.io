func guessNumber(n int) int {
	var binSearch func(l, r int) int
	binSearch = func(l, r int) int {
		mid := (l + r) / 2
		switch guess(mid) {
		case 0:
			return mid
		case -1:
			return binSearch(l, mid-1)
		case 1:
			return binSearch(mid+1, r)
		}
		return 0
	}
	return binSearch(1, n)
}
