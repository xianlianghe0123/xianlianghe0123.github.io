func findClosestElements(arr []int, k int, x int) []int {
	if len(arr) == k {
		return arr
	}
	j := sort.SearchInts(arr, x)
	i := j - 1
	for j-i <= k {
		if i == -1 {
			j++
			continue
		}
		if j == len(arr) {
			i--
			continue
		}
		t := arr[j] - x - (x - arr[i])
		if t >= 0 {
			i--
			continue
		}
		j++
	}
	ans := make([]int, 0, k)
	for i++; i < j; i++ {
		ans = append(ans, arr[i])
	}
	return ans
}
