func numUniqueEmails(emails []string) int {
	exist := make(map[string]struct{})
	for _, email := range emails {
		real := make([]byte, 0, len(email))
		t := strings.Split(email, "@")
	Outer:
		for _, char := range t[0] {
			switch char {
			case '.':
			case '+':
				break Outer
			default:
				real = append(real, byte(char))
			}
		}
		real = append(real, '@')
		real = append(real, t[1]...)
		exist[string(real)] = struct{}{}
	}
	return len(exist)
}
