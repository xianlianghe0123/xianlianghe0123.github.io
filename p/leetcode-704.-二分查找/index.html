func search(nums []int, target int) int {
	for l, r := 0, len(nums); l < r; {
		mid := (l + r) / 2
		if target == nums[mid] {
			return mid
		} else if target < nums[mid] {
			r = mid
		} else {
			l = mid + 1
		}
	}
	return -1
}
