func allPathsSourceTarget(graph [][]int) [][]int {
	res := make([][]int, 0, len(graph))
	path := make([]int, len(graph))
	var dfs func(i int, depth int)
	dfs = func(i int, depth int) {
		if depth == len(graph) {
			return
		}
		path[depth] = i
		// Âà∞ËææÁªàÁÇπÔºåÂä†ÂÖ•ÁªìÊûú
		if i == len(graph)-1 {
			t := make([]int, depth+1)
			copy(t, path[:depth+1])
			res = append(res, t)
			return
		}
		// ÈÅçÂéÜ‰∏ã‰∏Ä‰∏™ÁªìÁÇπ
		for _, next := range graph[i] {
			dfs(next, depth+1)
		}
	}
	dfs(0, 0)
	return res
}
