func maxFrequency(nums []int, k int) int {
	sort.Ints(nums)
	max := 1
	for l, r := 0, 1; r < len(nums); {
		need := (nums[r] - nums[r-1]) * (r - l)
		// Êâ©Â±ï
		if k >= need {
			k -= need
			r++
		} else { // Êî∂Áº©
			if r-1 > l {
				k += nums[r-1] - nums[l]
			}
			l++
		}
		if max < r-l {
			max = r - l
		}
	}
	return max
}
