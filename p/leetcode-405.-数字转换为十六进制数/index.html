func toHex(num int) string {
	if num == 0 {
		return "0"
	}
	digit := []byte("0123456789abcdef")
	ans := make([]byte, 0, 8)
	t := uint32(num)
	for ; t != 0; t /= 16 {
		ans = append(ans, digit[t%16])
	}
	for i := len(ans)/2 - 1; i >= 0; i-- {
		ans[i], ans[len(ans)-1-i] = ans[len(ans)-1-i], ans[i]
	}
	return string(ans)
}

func toHex(num int) string {
	if num == 0 {
		return "0"
	}
	digit := []byte("0123456789abcdef")
	ans := make([]byte, 0, 8)
	for i := 0; i < 8 && num != 0; i, num = i+1, num>>4 {
		ans = append(ans, digit[num&15])
	}
	for i := len(ans)/2 - 1; i >= 0; i-- {
		ans[i], ans[len(ans)-1-i] = ans[len(ans)-1-i], ans[i]
	}
	return string(ans)
}
