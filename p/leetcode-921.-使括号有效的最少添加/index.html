func minAddToMakeValid(s string) int {
	ans := 0
	left := 0
	for i := range s {
		if s[i] == '(' {
			left++
			continue
		}
		if left == 0 {
			ans++
		} else {
			left--
		}
	}
	ans += left
	return ans
}
