func getImportance(employees []*Employee, id int) int {
	m := make(map[int]*Employee)
	for _, e := range employees {
		m[e.Id] = e
	}
	var dfs func(id int) int
	dfs = func(id int) int {
		e, ok := m[id]
		if !ok {
			return 0
		}
		sum := e.Importance
		for _, s := range e.Subordinates {
			sum += dfs(s)
		}
		return sum
	}
	return dfs(id)
}
