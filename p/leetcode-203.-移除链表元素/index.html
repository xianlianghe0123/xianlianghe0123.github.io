func removeElements(head *ListNode, val int) *ListNode {
	head = &ListNode{
		Val:  0,
		Next: head,
	}
	for node := head; node.Next != nil; {
		if node.Next.Val == val {
			node.Next = node.Next.Next
		} else {
            node = node.Next
        }
	}
	head = head.Next
	return head
}
