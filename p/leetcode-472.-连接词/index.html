type Node struct {
	End      bool
	Children [26]*Node
}

func findAllConcatenatedWordsInADict(words []string) []string {
	sort.Slice(words, func(i, j int) bool {
		return len(words[i]) < len(words[j])
	})
	root := &Node{
		End:      false,
		Children: [26]*Node{},
	}
	var dfs func(word string) bool
	dfs = func(word string) bool {
		for node, i := root, 0; i < len(word); i++ {
			// Êó†Ê≥ïÁªßÁª≠ÂåπÈÖç
			node = node.Children[word[i]-'a']
			if node == nil {
				break
			}
			// ÂåπÈÖçÂà∞Êüê‰∏™ÂçïËØç
			if node.End {
				// ÂåπÈÖçÂà∞ÁªìÂ∞æÔºåÂåπÈÖçÊàêÂäü
				if i == len(word)-1 {
					return true
				}
				// Â∞ùËØïÂåπÈÖç‰∏ã‰∏Ä‰∏™
				if dfs(word[i+1:]) {
					return true
				}
			}
		}
		return false
	}
	ans := make([]string, 0, len(words))
	for _, word := range words {
		if dfs(word) {
			ans = append(ans, word)
			continue
		}
		// Ê∑ªÂä†Âà∞ TrieÊ†ë
		node := root
		for i := range word {
			if node.Children[word[i]-'a'] == nil {
				node.Children[word[i]-'a'] = &Node{
					End:      false,
					Children: [26]*Node{},
				}
			}
			node = node.Children[word[i]-'a']
		}
		node.End = true
	}
	return ans
}
