func heightChecker(heights []int) int {
	ans := 0
	sortedHeights := make([]int, len(heights))
	copy(sortedHeights, heights)
	sort.Ints(sortedHeights)
	for i := range heights {
		if heights[i] != sortedHeights[i] {
			ans++
		}
	}
	return ans
}
