func maxSlidingWindow(nums []int, k int) []int {
	if len(nums) == 0 {
		return nil
	}
	ans := make([]int, 0, len(nums)-k+1)
	for i := k; i <= len(nums); i++ {
		max := math.MinInt32
		for j := i - k; j < i; j++ {
			if max < nums[j] {
				max = nums[j]
			}
		}
		ans = append(ans, max)
	}
	return ans
}

		return nil

type MaxHeap struct {
	sort.IntSlice
}

func (this *MaxHeap) Less(i, j int) bool {
	return this.IntSlice.Less(j, i)
}

func (this *MaxHeap) Push(elem interface{}) {
	this.IntSlice = append(this.IntSlice, elem.(int))
}

func (this *MaxHeap) Pop() interface{} {
	elem := this.IntSlice[0]
	this.IntSlice = this.IntSlice[:this.Len()-1]
	return elem
}

func maxSlidingWindow(nums []int, k int) []int {
	if len(nums) == 0 {
		return nil
	}
	// åˆå§‹åŒ–
	h := &MaxHeap{
		IntSlice: make([]int, 0, len(nums)),
	}
	pos := make(map[int]int)
	for i := 0; i < k; i++ {
		pos[nums[i]] = i
		heap.Push(h, nums[i])
	}
	ans := make([]int, 0, len(nums)-k+1)
	ans = append(ans, h.IntSlice[0])
	for i := k; i < len(nums); i++ {
		pos[nums[i]] = i
		// å…¥å †
		heap.Push(h, nums[i])
		for pos[h.IntSlice[0]] <= i-k {
			heap.Pop(h)
		}
		ans = append(ans, h.IntSlice[0])
	}
	return ans
}
