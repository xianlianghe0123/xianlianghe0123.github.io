func openLock(deadends []string, target string) int {
	type Pair struct {
		Cur string
		Cnt int
	}

	deadendsMap := make(map[string]bool)
	for _, s := range deadends {
		deadendsMap[s] = true
	}
	if deadendsMap["0000"] {
		return -1
	}
	// å¹¿åº¦ä¼˜å…ˆæœç´¢
	min := -1
	queue := make([]*Pair, 0, 10000)
	queue = append(queue, &Pair{
		Cur: "0000",
		Cnt: 0,
	})
	deadendsMap["0000"] = true
	for len(queue) > 0 {
		cur := queue[0].Cur
		cnt := queue[0].Cnt
		queue = queue[1:]
		if cur == target {
			if min == -1 {
				min = cnt
				break
			}
		}
		for i := 0; i < 4; i++ {
			// å‘åæ—‹è½¬
			t := []byte(cur)
			t[i] = '0' + ((t[i]-'0')+1)%10
			if !deadendsMap[string(t)] {
				queue = append(queue, &Pair{
					Cur: string(t),
					Cnt: cnt + 1,
				})
				deadendsMap[string(t)] = true
			}
			// å‘å‰æ—‹è½¬
			t = []byte(cur)
			t[i] = '0' + ((t[i]-'0')-1+10)%10
			if !deadendsMap[string(t)] {
				queue = append(queue, &Pair{
					Cur: string(t),
					Cnt: cnt + 1,
				})
				deadendsMap[string(t)] = true
			}
		}
	}
	return min
}
