type StockSpanner struct {
	stack   [][]int
	counter int
}

func Constructor() StockSpanner {
	return StockSpanner{
		stack:   make([][]int, 0),
		counter: 0,
	}
}

func (this *StockSpanner) Next(price int) int {
	idx := -1
	for len(this.stack) != 0 {
		top := this.stack[len(this.stack)-1]
		if top[1] > price {
			idx = top[0]
			break
		}
		this.stack = this.stack[:len(this.stack)-1]
	}
	this.stack = append(this.stack, []int{this.counter, price})
	this.counter++
	return this.counter - 1 - idx
}
