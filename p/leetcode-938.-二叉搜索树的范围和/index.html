func rangeSumBST(root *TreeNode, low int, high int) int {
	sum := 0
	var search func(node *TreeNode)
	search = func(node *TreeNode) {
		if node == nil {
			return
		}
		if node.Val < low {
			search(node.Right)
		} else if node.Val > high {
			search(node.Left)
		} else {
			sum += node.Val
			search(node.Left)
			search(node.Right)
		}
	}
	search(root)
	return sum
}
