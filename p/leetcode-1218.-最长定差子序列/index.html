func longestSubsequence(arr []int, difference int) int {
	max := 0
	dp := make(map[int]int)
	for _, num := range arr {
		dp[num] = dp[num-difference] + 1
		if max < dp[num] {
			max = dp[num]
		}
	}
	return max
}
