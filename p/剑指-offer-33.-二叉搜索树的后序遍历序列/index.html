func verifyPostorder(postorder []int) bool {
	if len(postorder) <= 2 {
		return true
	}
	root := postorder[len(postorder)-1]
	split := len(postorder) - 2
	for ; split >= 0 && postorder[split] > root; split-- {
	}
	for i := 0; i < split; i++ {
		if postorder[i] > root {
			return false
		}
	}
	return verifyPostorder(postorder[:split+1]) && verifyPostorder(postorder[split+1:len(postorder)-1])
}
