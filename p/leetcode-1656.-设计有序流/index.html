type OrderedStream struct {
	elems []string
	id    int
}

func Constructor(n int) OrderedStream {
	return OrderedStream{
		elems: make([]string, n),
		id:    0,
	}
}

func (this *OrderedStream) Insert(idKey int, value string) []string {
	this.elems[idKey-1] = value
	start := this.id
	for ; this.id < len(this.elems); this.id++ {
		if this.elems[this.id] == "" {
			break
		}
	}
	ans := make([]string, this.id-start)
	copy(ans, this.elems[start:this.id])
	return ans
}
