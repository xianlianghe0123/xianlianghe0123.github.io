func tree2str(root *TreeNode) string {
	ans := make([]byte, 0, 10)
	var dfs func(node *TreeNode)
	dfs = func(node *TreeNode) {
		if node == nil {
			return
		}
		ans = append(ans, strconv.Itoa(node.Val)...)
		if node.Left != nil || node.Right != nil {
			ans = append(ans, '(')
			dfs(node.Left)
			ans = append(ans, ')')
		}
		if node.Right != nil {
			ans = append(ans, '(')
			dfs(node.Right)
			ans = append(ans, ')')
		}
	}
	dfs(root)
	return string(ans)
}
