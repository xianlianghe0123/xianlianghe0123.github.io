func imageSmoother(img [][]int) [][]int {
	ans := make([][]int, len(img))
	for row, line := range img {
		ans[row] = make([]int, len(line))
		for col := range line {
			sum := 0 // Ê±ÇÂíå
			cnt := 0 // ÂçïÂÖÉÊ†ºÊï∞Èáè
			// Ëá™Ë∫´&Âë®Âõ¥ÂçïÂÖÉÊ†ºÊ±ÇÂíå
			for i := row - 1; i <= row+1; i++ {
				for j := col - 1; j <= col+1; j++ {
					if i < 0 || i >= len(img) || j < 0 || j >= len(line) {
						continue
					}
					sum += img[i][j]
					cnt++
				}
			}
			ans[row][col] = sum / cnt // Âπ≥ÂùáÂÄº
		}
	}
	return ans
}
