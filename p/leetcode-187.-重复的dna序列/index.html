func findRepeatedDnaSequences(s string) []string {
	ans := make([]string, 0)
	cnt := make(map[string]int)
	for i := 10; i <= len(s); i++ {
		cnt[s[i-10:i]]++
		if cnt[s[i-10:i]] == 2 {
			ans = append(ans, s[i-10:i])
		}
	}
	return ans
}
