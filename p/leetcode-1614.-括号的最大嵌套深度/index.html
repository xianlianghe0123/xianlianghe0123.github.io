func maxDepth(s string) int {
	maxDepth := 0
	depth := 0
	for i := range s {
		switch s[i] {
		case '(':
			depth++
			if maxDepth < depth {
				maxDepth = depth
			}
		case ')':
			depth--
		}
	}
	return maxDepth
}
