func findPeakElement(nums []int) int {
	for i, num := range nums {
		if (i == 0 || (i > 0 && num > nums[i-1])) && (i == len(nums)-1 || (i < len(nums)-1 && num > nums[i+1])) {
			return i
		}
	}
	return 0
}

func findPeakElement(nums []int) int {
	for l, r := 0, len(nums); l < r; {
		if l+1 == r {
			return l
		}
		mid := (l + r) / 2
		if mid == 0 || (mid > 0 && nums[mid] > nums[mid-1]) {
			l = mid
		} else {
			r = mid
		}
	}
	return 0
}
