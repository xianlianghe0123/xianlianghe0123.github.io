func constructArr(a []int) []int {
	if len(a) == 0 {
		return make([]int, 0)
	}
	ans := make([]int, len(a))
	ans[0] = 1
	for i, prefixProduct := 1, 1; i < len(a); i++ {
		prefixProduct *= a[i-1]
		ans[i] = prefixProduct
	}
	for i, suffixProduct := len(a)-2, 1; i >= 0; i-- {
		suffixProduct *= a[i+1]
		ans[i] *= suffixProduct
	}
	return ans
}
