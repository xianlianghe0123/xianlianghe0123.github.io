func countKDifference(nums []int, k int) int {
	ans := 0
	for i := len(nums) - 2; i >= 0; i-- {
		for j := i + 1; j < len(nums); j++ {
			if nums[i]-nums[j] == k || nums[j]-nums[i] == k {
				ans++
			}
		}
	}
	return ans
}

func countKDifference(nums []int, k int) int {
	ans := 0
	cnt := make(map[int]int)
	for _, num := range nums {
		ans += cnt[num-k]
		ans += cnt[num+k]
		cnt[num]++
	}
	return ans
}
