func originalDigits(s string) string {
	mapping := map[byte]struct {
		en  string
		num byte
	}{
		'z': {en: "zero", num: '0'},
		'o': {en: "one", num: '1'},
		'w': {en: "two", num: '2'},
		't': {en: "three", num: '3'},
		'u': {en: "four", num: '4'},
		'f': {en: "five", num: '5'},
		'x': {en: "six", num: '6'},
		's': {en: "seven", num: '7'},
		'g': {en: "eight", num: '8'},
		'e': {en: "nine", num: '9'},
	}
	// ÁªüËÆ°Â≠óÁ¨¶Êï∞Èáè
	charCnt := make(map[byte]int)
	for i := range s {
		charCnt[s[i]]++
	}

	numCnt := make(map[byte]int)
	length := 0
	for _, char := range []byte("zwuxgotfse") {
		if charCnt[char] <= 0 {
			continue
		}
		numCnt[mapping[char].num] = charCnt[char]
		length += charCnt[char]
		// Âà†Èô§Â∑≤Áü•Êï∞Â≠óÂ≠óÁ¨¶
		for i := range mapping[char].en {
			charCnt[mapping[char].en[i]] -= numCnt[mapping[char].num]
		}
	}
	// ÊûÑÈÄ†ÁªìÊûú
	ans := make([]byte, 0, length)
	for char := byte('0'); char <= '9'; char++ {
		for j := numCnt[char]; j > 0; j-- {
			ans = append(ans, char)
		}
	}
	return string(ans)
}
