func removeOuterParentheses(s string) string {
	depth := 0
	ans := make([]byte, 0, len(s))
	for i := range s {
		switch s[i] {
		case '(':
			depth++
			if depth == 1 {
				continue
			}
		case ')':
			depth--
			if depth == 0 {
				continue
			}
		}
		ans = append(ans, s[i])
	}
	return string(ans)
}
