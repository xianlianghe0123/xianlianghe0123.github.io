func removeOuterParentheses(s string) string {
	stack := make([]byte, 0, len(s))
	ans := make([]byte, 0, len(s))
	for i := range s {
		switch s[i] {
		case '(':
			stack = append(stack, s[i])
			// Ê†àÂ∫ï Ôºà ÂøΩÁï•

		case ')':
			stack = stack[:len(stack)-1] // ÈÖçÂØπ

		}
		ans = append(ans, s[i])
	}
	return string(ans)
}

func removeOuterParentheses(s string) string {
	depth := 0
	ans := make([]byte, 0, len(s))
	for i := range s {
		switch s[i] {
		case '(':
			depth++
			if depth == 1 {
				continue
			}
		case ')':
			depth--
			if depth == 0 {
				continue
			}
		}
		ans = append(ans, s[i])
	}
	return string(ans)
}
