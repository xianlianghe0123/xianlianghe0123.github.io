func countConsistentStrings(allowed string, words []string) int {
	ans := 0
	mask := 0
	for _, a := range allowed {
		mask |= 1 << (a - 'a')
	}
	for _, word := range words {
		exist := true
		for _, char := range word {
			if (1<<(char-'a'))&mask == 0 {
				exist = false
				break
			}
		}
		if exist {
			ans++
		}
	}
	return ans
}
