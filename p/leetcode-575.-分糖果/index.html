func distributeCandies(candyType []int) int {
	types := make(map[int]struct{})
	for _, t := range candyType {
		types[t] = struct{}{}
	}
	if len(candyType)/2 < len(types) {
		return len(candyType) / 2
	} else {
		return len(types)
	}
}
