func stringMatching(words []string) []string {
	ans := make([]string, 0, len(words))
	for i := range words {
		for j := range words {
			if i == j {
				continue
			}
			if strings.Contains(words[j], words[i]) {
				ans = append(ans, words[i])
				break
			}
		}
	}
	return ans
}
