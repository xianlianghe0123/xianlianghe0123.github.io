func wiggleSort(nums []int) {
	dup := make([]int, len(nums))
	copy(dup, nums)
	sort.Ints(dup)
	l, r := (len(nums)-1)/2, len(nums)-1
	for i := range nums {
		if i&1 == 0 {
			nums[i] = dup[l]
			l--
		} else {
			nums[i] = dup[r]
			r--
		}
	}
}
