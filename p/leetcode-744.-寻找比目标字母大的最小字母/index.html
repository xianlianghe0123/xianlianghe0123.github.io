func nextGreatestLetter(letters []byte, target byte) byte {
	l, r := 0, len(letters)-1
	for l < r { // Áõ¥Âà∞ l == r ‰∏∫Ê≠¢
		mid := (l + r) / 2
		if letters[mid] <= target {
			l = mid + 1
		} else {
			r = mid
		}
	}
	if letters[l] > target {
		return letters[l]
	}
	return letters[0]
}
