func advantageCount(nums1 []int, nums2 []int) []int {
	n := len(nums1)
	idx2 := make([]int, 0, n)
	for i := 0; i < n; i++ {
		idx2 = append(idx2, i)
	}
	sort.Ints(nums1)
	sort.Slice(idx2, func(i, j int) bool {
		return nums2[idx2[i]] < nums2[idx2[j]]
	})

	ans := make([]int, n)
	j, k := 0, n-1
	for _, num1 := range nums1 {
		if num1 <= nums2[idx2[j]] {
			ans[idx2[k]] = num1
			k--
			continue
		}
		ans[idx2[j]] = num1
		j++
	}
	return ans
}
