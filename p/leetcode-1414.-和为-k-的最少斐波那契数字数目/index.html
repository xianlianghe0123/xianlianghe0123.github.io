func findMinFibonacciNumbers(k int) int {
	f := []int{1, 1}
	for f[len(f)-1] < k {
		f = append(f, f[len(f)-1]+f[len(f)-2])
	}
	cnt := 0
	for i := len(f) - 1; i >= 0; i-- {
		if k < f[i] {
			continue
		}
		k -= f[i]
		cnt++
		if k == 0 {
			break
		}
	}
	return cnt
}
