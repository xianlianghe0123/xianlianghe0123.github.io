func optimalDivision(nums []int) string {
	type Node struct {
		max         float64
		min         float64
		maxEquation string
		minEquation string
	}
	dp := make([][]*Node, len(nums))
	// åˆ†æ¯éå•ä¸ªæ•°å­—ï¼Œéœ€è¦åŠ æ‹¬å·
	format := func(need bool, dividend, divisor string) string {
		if need {
			return fmt.Sprintf("%s/(%s)", dividend, divisor)
		}
		return fmt.Sprintf("%s/%s", dividend, divisor)
	}
	for i := len(nums) - 1; i >= 0; i-- {
		dp[i] = make([]*Node, len(nums))
		dp[i][i] = &Node{
			max:         float64(nums[i]),
			min:         float64(nums[i]),
			maxEquation: strconv.Itoa(nums[i]),
			minEquation: strconv.Itoa(nums[i]),
		}
		for j := i + 1; j < len(nums); j++ {
			node := &Node{
				min: math.MaxFloat64,
			}
			for k := i; k < j; k++ {
				// æœ€å¤§å€¼ï¼Œè¢«é™¤æ•°æœ€å¤§ï¼Œé™¤æ•°æœ€å°
				if node.max < dp[i][k].max/dp[k+1][j].min {
					node.max = dp[i][k].max / dp[k+1][j].min
					node.maxEquation = format(j-k > 1, dp[i][k].maxEquation, dp[k+1][j].minEquation)
				}
				// æœ€å°å€¼ï¼Œè¢«é™¤æ•°æœ€å°ï¼Œé™¤æ•°æœ€å¤§
				if node.min > dp[i][k].min/dp[k+1][j].max {
					node.min = dp[i][k].min / dp[k+1][j].max
					node.minEquation = format(j-k > 1, dp[i][k].minEquation, dp[k+1][j].maxEquation)
				}
			}
			dp[i][j] = node
		}
	}
	return dp[0][len(nums)-1].maxEquation
}
