func maxIncreaseKeepingSkyline(grid [][]int) int {
	rowMax := make([]int, len(grid))
	colMax := make([]int, len(grid[0]))
	for i, row := range grid {
		for j, cell := range row {
			rowMax[i] = Max(rowMax[i], cell) // Á¨¨ i Ë°åÊúÄÂ§ßÂÄº
			colMax[j] = Max(colMax[j], cell) // Á¨¨ j ÂàóÊúÄÂ§ßÂÄº
		}
	}
	ans := 0
	for i, row := range grid {
		for j, cell := range row {
			ans += Min(rowMax[i], colMax[j]) - cell // ËÆ°ÁÆóÂèØ‰ª•Â¢ûÂä†ÁöÑÈ´òÂ∫¶
		}
	}
	return ans
}

func Max(x, y int) int {
	if x > y {
		return x
	}
	return y
}

func Min(x, y int) int {
	if x < y {
		return x
	}
	return y
}
