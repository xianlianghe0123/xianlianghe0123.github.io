func spiralOrder(matrix [][]int) []int {
	if len(matrix) == 0 {
		return make([]int, 0)
	}
	visit := make([][]bool, len(matrix))
	for i := range matrix {
		visit[i] = make([]bool, len(matrix[i]))
	}
	ans := make([]int, 0, len(matrix)*len(matrix[0]))
	directions := [4][2]int{{0, 1}, {1, 0}, {0, -1}, {-1, 0}}
	for i, j, direction := 0, 0, 0; len(ans) < cap(ans); i, j = i+directions[direction][0], j+directions[direction][1] {
		ans = append(ans, matrix[i][j])
		visit[i][j] = true
        // éå†æ–¹å‘ä¿®æ­£
		nextI, nextJ := i+directions[direction][0], j+directions[direction][1]
		if nextI < 0 || nextI >= len(matrix) || nextJ < 0 || nextJ >= len(matrix[0]) || visit[nextI][nextJ] {
			direction = (direction + 1) % len(directions)
		}
	}
	return ans
}

func spiralOrder(matrix [][]int) []int {
	if len(matrix) == 0 {
		return make([]int, 0)
	}
	ans := make([]int, 0, len(matrix)*len(matrix[0]))
	for up, down, left, right := 0, len(matrix)-1, 0, len(matrix[0])-1; up <= down && left <= right; up, down, left, right = up+1, down-1, left+1, right-1 {
		for i := left; i <= right; i++ {
			ans = append(ans, matrix[up][i])
		}
		for i := up + 1; i <= down; i++ {
			ans = append(ans, matrix[i][right])
		}
		if up != down && left != right {
			for i := right - 1; i >= left; i-- {
				ans = append(ans, matrix[down][i])
			}
			for i := down - 1; i > up; i-- {
				ans = append(ans, matrix[i][left])
			}
		}
	}
	return ans
}
