func orderlyQueue(s string, k int) string {
	if k > 1 {
		bytes := []byte(s)
		sort.Slice(bytes, func(i, j int) bool {
			return bytes[i] < bytes[j]
		})
		return string(bytes)
	}
	minIdx := 0
	for i := 1; i < len(s); i++ {
		// ÊØîËæÉ‰∏§‰∏™Â≠óÁ¨¶‰∏≤
		for j := 0; j < len(s); j++ {
			if s[(minIdx+j)%len(s)] < s[(i+j)%len(s)] {
				break
			} else if s[(minIdx+j)%len(s)] > s[(i+j)%len(s)] {
				minIdx = i
				break
			}
		}
	}
	return s[minIdx:] + s[:minIdx]
}
