func reorderList(head *ListNode) {
	length := 0
	for node := head; node != nil; node = node.Next {
		length++
	}
	list := make([]*ListNode, 0, length)
	for node := head; node != nil; node = node.Next {
		list = append(list, node)
	}
	ans := new(ListNode)
	node := ans
	i := 0
	for ; i < length/2; i++ {
		node.Next = list[i]
		node = node.Next
		node.Next = list[length-1-i]
		node = node.Next
	}
	if length&1 == 1 {
		node.Next = list[i]
		node = node.Next
	}
	node.Next = nil
	head = ans.Next
}

func reorderList(head *ListNode) {
	length := 0
	for node := head; node != nil; node = node.Next {
		length++
	}
	if length <= 1 {
		return
	}
	// å‰åŠéƒ¨åˆ†è·³è¿‡
	node := head
	for i := length/2 - 1; i > 0; i-- {
		node = node.Next
	}
	node.Next, node = nil, node.Next // æˆªæ–­
	// ååŠéƒ¨åˆ†åè½¬
	last := (*ListNode)(nil)
	for node != nil {
		node.Next, node, last = last, node.Next, node
	}
	tail := new(ListNode)
	for node = head; node != nil; {
		tail.Next = node
		tail = tail.Next
		node = node.Next
		tail.Next = last
		tail = tail.Next
		last = last.Next
	}
	if last != nil {
		tail.Next = last
	}
}
