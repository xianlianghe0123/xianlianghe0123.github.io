func calPoints(ops []string) int {
	sum := 0
	stack := make([]int, 0, len(ops))
	for _, op := range ops {
		num := 0
		switch op {
		case "C":
			sum -= stack[len(stack)-1]   // ‰∏ä‰∏Ä‰∏™Êï∞‰ªéÂíå‰∏≠ÂéªÈô§

		case "+":
			num = stack[len(stack)-1] + stack[len(stack)-2]
		case "D":
			num = stack[len(stack)-1] << 1
		default:
			num, _ = strconv.Atoi(op)
		}
		sum += num                 // Ê±ÇÂíå
		stack = append(stack, num) // ÂÖ•Ê†à
	}
	return sum
}
