func maxAscendingSum(nums []int) int {
	ans := 0
	sum := nums[0]
	for i := 1; i < len(nums); i++ {
		if nums[i] > nums[i-1] {
			sum += nums[i]
			continue
		}
		if ans < sum {
			ans = sum
		}
		sum = nums[i]
	}
	if ans < sum {
		ans = sum
	}
	return ans
}
