func isValidSudoku(board [][]byte) bool {
	for i := 0; i < 9; i++ {
		var (
			rowExist    [9]bool
			colExist    [9]bool
			SquareExist [9]bool
		)
		for j := 0; j < 9; j++ {
			if board[i][j] != '.' {
				if rowExist[board[i][j]-'1'] {
					return false
				}
				rowExist[board[i][j]-'1'] = true
			}
			if board[j][i] != '.' {
				if colExist[board[j][i]-'1'] {
					return false
				}
				colExist[board[j][i]-'1'] = true
			}
			if board[i/3*3+j/3][i%3*3+j%3] != '.' {
				if SquareExist[board[i/3*3+j/3][i%3*3+j%3]-'1'] {
					return false
				}
				SquareExist[board[i/3*3+j/3][i%3*3+j%3]-'1'] = true
			}
		}
	}
	return true
}
