func search(nums []int, target int) int {
	pos := len(nums)
	for l, r := 0, len(nums)-1; l <= r; {
		if l == r {
			pos = l
			break
		}
		mid := (l + r) / 2
		if nums[mid] < target {
			l = mid + 1
		} else {
			r = mid
		}
	}
	cnt := 0
	for i := pos; i < len(nums) && nums[i] == target; i++ {
		cnt++
	}
	return cnt
}
