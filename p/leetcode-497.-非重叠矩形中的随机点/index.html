type Solution struct {
	rects  [][]int
	sum    []int
	random *rand.Rand
}

func Constructor(rects [][]int) Solution {
	sum := make([]int, len(rects)+1)
	for i, rect := range rects {
		sum[i+1] = sum[i] + (rect[2]-rect[0]+1)*(rect[3]-rect[1]+1)
	}
	return Solution{
		rects:  rects,
		sum:    sum,
		random: rand.New(rand.NewSource(time.Now().UnixNano())),
	}
}

func (this *Solution) Pick() []int {
	pointSeq := this.random.Intn(this.sum[len(this.sum)-1])
	i := sort.SearchInts(this.sum, pointSeq+1) // Á°ÆÂÆöÁü©ÂΩ¢
	pointSeq -= this.sum[i-1]                  // ÂΩìÂâçÁü©ÂΩ¢Á¨¨ pointSeq ‰∏™ÁÇπ
	fmt.Println(pointSeq)
	rect := this.rects[i-1]
	return []int{rect[0] + pointSeq%(rect[2]-rect[0]+1), rect[1] + pointSeq/(rect[2]-rect[0]+1)}
}
