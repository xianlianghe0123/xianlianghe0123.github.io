func projectionArea(grid [][]int) int {
	xy := 0
	xz := 0
	yz := 0
	for x := 0; x < len(grid); x++ {
		xzMaxZ := 0 // Áõ∏ÂêåxÂùêÊ†áÔºåzÂùêÊ†áÊúÄÂ§ßÂÄº
		yzMaxZ := 0 // Áõ∏ÂêåyÂùêÊ†áÔºåzÂùêÊ†áÊúÄÂ§ßÂÄº
		for y := 0; y < len(grid); y++ {
			if grid[x][y] != 0 {
				xy++
			}
			if xzMaxZ < grid[x][y] {
				xzMaxZ = grid[x][y]
			}
			if yzMaxZ < grid[y][x] {
				yzMaxZ = grid[y][x]
			}
		}
		xz += xzMaxZ
		yz += yzMaxZ
	}
	return xy + xz + yz
}
