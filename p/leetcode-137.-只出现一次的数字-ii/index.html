func singleNumber(nums []int) int {
	cnt := make(map[int]int)
	for _, num := range nums {
		cnt[num]++
	}
	for k, v := range cnt {
		if v == 1 {
			return k
		}
	}
	return 0
}

func singleNumber(nums []int) int {
	sort.Ints(nums)
	res := nums[0]
	for i := 2; i < len(nums); i += 3 {
		if res == nums[i] {
			res = nums[i+1]
			continue
		} else {
			break
		}
	}
	return res
}

func singleNumber(nums []int) int {
	a, b := 0, 0
	for _, x := range nums {
		t := ^a&b&x | a&^b&^x
		b = ^a & (b ^ x)
		a = t
	}
	return b
}
