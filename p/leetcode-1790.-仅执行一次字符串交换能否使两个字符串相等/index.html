func areAlmostEqual(s1 string, s2 string) bool {
	diff := make([]int, 0, 2)
	for i := range s1 {
		if s1[i] == s2[i] {
			continue
		}
		// è¶…è¿‡ä¸¤å¤„ä¸åŒ
		if len(diff) == 2 {
			return false
		}
		diff = append(diff, i)
	}
	if len(diff) == 0 {
		return true
	}
	if len(diff) == 1 {
		return false
	}
	if s1[diff[0]] == s2[diff[1]] && s1[diff[1]] == s2[diff[0]] {
		return true
	}
	return false
}
