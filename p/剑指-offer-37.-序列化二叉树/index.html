class Codec:

    def serialize(self, root):
        """Encodes a tree to a single string.

        :type root: TreeNode
        :rtype: str
        """
        list = []

        def dfs(node):
            if node == None:
                list.append("None")
            else:
                list.append(str(node.val))
                dfs(node.left)
                dfs(node.right)

        dfs(root)
        return ",".join(list)


    def deserialize(self, data):
        """Decodes your encoded data to tree.

        :type data: str
        :rtype: TreeNode
        """
        list = data.split(",")
        i = 0
        def dfs():
            nonlocal i
            if list[i] == "None":
                i += 1
                return None
            node = TreeNode(int(list[i]))
            i += 1
            node.left = dfs()
            node.right = dfs()

            return node

        return dfs()
