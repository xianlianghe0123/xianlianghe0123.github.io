func sumSubseqWidths(nums []int) int {
	const Mod = int(1e9 + 7)
	sort.Ints(nums)
	sum := 0
	for j, t, power := 1, 0, 1; j < len(nums); j++ {
		t = (t*2 + nums[j-1]) % Mod
		power = (power * 2) % Mod
		sum = (sum + nums[j]*(power-1) - t) % Mod
	}
	return sum
}
