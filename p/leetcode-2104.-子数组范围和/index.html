func subArrayRanges(nums []int) int64 {
	sum := int64(0)
	for i := len(nums) - 2; i >= 0; i-- {
		min := nums[i]
		max := nums[i]
		for j := i + 1; j < len(nums); j++ {
			if max < nums[j] {
				max = nums[j]
			}
			if min > nums[j] {
				min = nums[j]
			}
			sum += int64(max - min)
		}
	}
	return sum
}
