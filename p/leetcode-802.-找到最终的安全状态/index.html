func eventualSafeNodes(graph [][]int) []int {
	const (
		ALREADY_VISIT = -1
		NOT_VISIT     = 0
		VISIT         = 1
		ARRIVE_RING   = 2
	)
	attrs := make([]int, len(graph))
	var dfs func(node int) bool
	dfs = func(node int) bool {
		switch attrs[node] {
		case ALREADY_VISIT:
			return false
		case NOT_VISIT:
			attrs[node] = VISIT
		case VISIT:
			attrs[node] = ARRIVE_RING
			return true
		case ARRIVE_RING:
			return true
		}
		// ÈÅçÂéÜÂêéÁª≠ÁªìÁÇπÔºåÊòØÂê¶ÂèØ‰ª•ÂèØÂà∞ËææÁéØ
		arriveRing := false
		for _, next := range graph[node] {
			arriveRing = arriveRing || dfs(next)
		}
		// ÂêéÁª≠ÊúâÁªìÁÇπÂèØÂà∞ËææÁéØÔºåÂàôÂΩìÂâçÁªìÁÇπ‰πüÂèØ‰ª•Âà∞ËææÁéØ
		if arriveRing {
			attrs[node] = ARRIVE_RING
		}
		// Â∑≤ËÆøÈóÆËøáÁöÑÁªìÁÇπÔºåÈò≤Ê≠¢ÈáçÂ§çÈÅçÂéÜ
		if attrs[node] == VISIT {
			attrs[node] = ALREADY_VISIT
		}
		return arriveRing
	}
	// dfs ÈÅçÂéÜÂõæ
	for i := 0; i < len(graph); i++ {
		if attrs[i] == NOT_VISIT {
			dfs(i)
		}
	}
	// ÈÅçÂéÜÂØªÊâæÂÆâÂÖ®ÁªìÁÇπ
	res := make([]int, 0, len(graph))
	for i, attr := range attrs {
		if attr != ARRIVE_RING {
			res = append(res, i)
		}
	}
	return res
}
