func eventualSafeNodes(graph [][]int) []int {
	const (
		ALREADY_VISIT = -1
		NOT_VISIT     = 0
		VISIT         = 1
		ARRIVE_RING   = 2
	)
	attrs := make([]int, len(graph))
	var dfs func(node int) bool
	dfs = func(node int) bool {
		switch attrs[node] {
		case ALREADY_VISIT:
			return false
		case NOT_VISIT:
			attrs[node] = VISIT
		case VISIT:
			attrs[node] = ARRIVE_RING
			return true
		case ARRIVE_RING:
			return true
		}
		// éå†åç»­ç»“ç‚¹ï¼Œæ˜¯å¦å¯ä»¥å¯åˆ°è¾¾ç¯
		arriveRing := false
		for _, next := range graph[node] {
			arriveRing = arriveRing || dfs(next)
		}
		// åç»­æœ‰ç»“ç‚¹å¯åˆ°è¾¾ç¯ï¼Œåˆ™å½“å‰ç»“ç‚¹ä¹Ÿå¯ä»¥åˆ°è¾¾ç¯
		if arriveRing {

		}

		if attrs[node] == VISIT {

		}

	}

		}

		}
	}
	return res
}
