func findMaxForm(strs []string, m int, n int) int {
	dp := [101][101]int{}
	for _, str := range strs {
		zeros := 0
		ones := 0
		for i := 0; i < len(str); i++ {
			if str[i] == '0' {
				zeros++
			} else {
				ones++
			}
		}
		for j := m; j >= 0; j-- {
			for k := n; k >= 0; k-- {
				if j >= zeros && k >= ones && dp[j-zeros][k-ones] >= dp[j][k] {
					dp[j][k] = dp[j-zeros][k-ones] + 1
				}
			}
		}
	}
	return dp[m][n]
}
