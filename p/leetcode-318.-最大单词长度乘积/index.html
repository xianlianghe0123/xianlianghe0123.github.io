func maxProduct(words []string) int {
	max := 0
	for i := len(words) - 1; i > 0; i-- {
		exist := make([]bool, 26)
		for k := range words[i] {
			exist[words[i][k]-'a'] = true
		}
	Other:
		for j := i - 1; j >= 0; j-- {
			for k := range words[j] {
				if exist[words[j][k]-'a'] {
					continue Other
				}
			}
			if max < len(words[i])*len(words[j]) {
				max = len(words[i]) * len(words[j])
			}
		}
	}
	return max
}

func maxProduct(words []string) int {
	sort.Slice(words, func(i, j int) bool {
		return len(words[i]) < len(words[j])
	})
	max := 0
	for i := len(words) - 1; i > 0; i-- {
		if len(words[i])*len(words[i]) < len(words[len(words)-1]) {
			break
		}
		exist := make([]bool, 26)
		for k := range words[i] {
			exist[words[i][k]-'a'] = true
		}
	Other:
		for j := i - 1; j >= 0; j-- {
			if len(words[i])*len(words[j]) < max {
				break
			}
			for k := range words[j] {
				if exist[words[j][k]-'a'] {
					continue Other
				}
			}
			if max < len(words[i])*len(words[j]) {
				max = len(words[i]) * len(words[j])
				break
			}
		}
	}
	return max
}

func maxProduct(words []string) int {
	chars := make([]int32, len(words))
	max := 0
	for i, word := range words {
		for j := range word {
			chars[i] |= 1 << (word[j] - 'a')
		}
	}
	for i := len(words) - 1; i > 0; i-- {
		for j := i - 1; j >= 0; j-- {
			if chars[i]&chars[j] == 0 && max < len(words[i])*len(words[j]) {
				max = len(words[i]) * len(words[j])
			}
		}
	}
	return max
}
