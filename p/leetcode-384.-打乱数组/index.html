type Solution struct {
	origin []int
	rand   *rand.Rand
}

func Constructor(nums []int) Solution {
	return Solution{
		origin: nums,
		rand:   rand.New(rand.NewSource(time.Now().UnixNano())),
	}
}

func (this *Solution) Reset() []int {
	return this.origin
}

func (this *Solution) Shuffle() []int {
	shuffle := make([]int, len(this.origin))
	bak := make([]int, len(this.origin))
	copy(bak, this.origin)
	for i := range this.origin {
		idx := this.rand.Intn(len(bak))
		shuffle[i] = bak[idx]
		bak = append(bak[:idx], bak[idx+1:]...)
	}
	return shuffle
}

type Solution struct {
	origin []int
	rand   *rand.Rand
}

func Constructor(nums []int) Solution {
	return Solution{
		origin: nums,
		rand:   rand.New(rand.NewSource(time.Now().UnixNano())),
	}
}

func (this *Solution) Reset() []int {
	return this.origin
}

func (this *Solution) Shuffle() []int {
	shuffle := make([]int, len(this.origin))
	copy(shuffle, this.origin)
	for i := range shuffle {
		idx := this.rand.Intn(len(shuffle) - i)
		shuffle[idx], shuffle[len(shuffle)-1-i] = shuffle[len(shuffle)-1-i], shuffle[idx]
	}
	return shuffle
}
