func maxValue(grid [][]int) int {
	dp := make([][]int, len(grid))
	for i := 0; i < len(grid); i++ {
		dp[i] = make([]int, len(grid[0]))
		for j := 0; j < len(grid[0]); j++ {
			if i-1 >= 0 {
				dp[i][j] = dp[i-1][j]
			}
			if j-1 >= 0 && dp[i][j] < dp[i][j-1] {
				dp[i][j] = dp[i][j-1]
			}
			dp[i][j] += grid[i][j]
		}
	}
	return dp[len(dp)-1][len(dp[0])-1]
}
