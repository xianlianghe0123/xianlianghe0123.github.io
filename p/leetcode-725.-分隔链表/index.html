func splitListToParts(head *ListNode, k int) []*ListNode {
    length := 0
    for node := head; node != nil; node = node.Next {
        length++
    }
    ans := make([]*ListNode, 0, k)
    for ;k != 0; k-- {
        split := length/k
        if length%k != 0 {
            split++
        }
        length -= split
        ans = append(ans, head)
        for i := 0; i < split && head != nil ; i++ {
            t := head
            head = head.Next
            if i+1 == split {
                t.Next = nil
            }
        }
    }
    return ans
}
