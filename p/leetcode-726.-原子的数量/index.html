func countOfAtoms(formula string) string {
	// Ëé∑Âèñ‰∏ã‰∏Ä‰∏™Êï∞Â≠ó
	getNum := func(idx *int) int {
		num := 0
		for ; *idx < len(formula); *idx++ {
			if formula[*idx] >= '0' && formula[*idx] <= '9' {
				num = num*10 + int(formula[*idx]-'0')
			} else {
				break
			}
		}
		if num == 0 {
			num = 1
		}
		return num
	}
	// Ëé∑Âèñ‰∏ã‰∏Ä‰∏™ÂÖÉÁ¥†
	getAtom := func(idx *int) string {
		atom := make([]byte, 0, 2)
		atom = append(atom, formula[*idx])
		for *idx++; *idx < len(formula); *idx++ {
			if formula[*idx] >= 'a' && formula[*idx] <= 'z' {
				atom = append(atom, formula[*idx])
			} else {
				break
			}
		}
		return string(atom)
	}
	// Ê†à
	stack := make([]map[string]int, 0, len(formula))
	// ÂÖ•Ê†à
	merge := func(tmp *map[string]int) {
		if len(*tmp) == 0 {
			return
		}
		if len(stack) > 0 && len(stack[len(stack)-1]) > 0 {
			for k, v := range *tmp {
				stack[len(stack)-1][k] += v
			}
		} else {
			stack = append(stack, *tmp)
		}
		*tmp = make(map[string]int)
	}
	// ËÆ°Êï∞map
	tmp := make(map[string]int)
	for i := 0; i < len(formula); {
		switch formula[i] {
		// Â∑¶Êã¨Âè∑ÊâÄÊúâÂÖÉÁ¥†ÂÖ•Ê†à
		case '(':
			merge(&tmp)
			stack = append(stack, nil)
			i++
		// Âè≥Êã¨Âè∑ÔºåÊúÄËøëÂ∑¶Êã¨Âè∑ËåÉÂõ¥ÁöÑÂÖÉÁ¥†Êï∞Èáè ‰πò‰ª• Êã¨Âè∑ÂêéÁöÑÊï∞
		case ')':
			i++
			merge(&tmp)
			cnt := getNum(&i)
			for len(stack) > 0 {
				pairs := stack[len(stack)-1]
				stack = stack[:len(stack)-1]
				if pairs == nil {
					break
				}
				for k, v := range pairs {
					tmp[k] += v * cnt
				}
			}
			merge(&tmp)
		default:
			tmp[getAtom(&i)] += getNum(&i)
		}
	}
	// ÂêàÂπ∂‰∏∫‰∏Ä‰∏™map
	merge(&tmp)
	tmp = make(map[string]int)
	for _, s := range stack {
		for atom, cnt := range s {
			tmp[atom] += cnt
		}
	}
	// Â≠óÂÖ∏Â∫èÊéíÂ∫è
	atoms := make([]string, 0, len(tmp))
	for atom := range tmp {
		atoms = append(atoms, atom)
	}
	sort.Strings(atoms)
	// ÁîüÊàêÁªìÊûúÂ≠óÁ¨¶‰∏≤
	res := make([]byte, 0, len(formula)<<1)
	for _, atom := range atoms {
		res = append(res, []byte(atom)...)
		if tmp[atom] > 1 {
			res = append(res, []byte(strconv.FormatInt(int64(tmp[atom]), 10))...)
		}
	}
	return string(res)
}
