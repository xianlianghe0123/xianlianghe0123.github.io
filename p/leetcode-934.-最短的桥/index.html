var dirs = [][]int{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}

func shortestBridge(grid [][]int) int {
	row, col := func() (int, int) {
		for row := range grid {
			for col := range grid[row] {
				if grid[row][col] == 1 {
					return row, col
				}
			}
		}
		return 0, 0
	}()
	// ÂØªÊâæËæπÁïå
	borders := list.New()

	q := list.New()
	q.PushBack([]int{row, col})
	grid[row][col] = -1
	for q.Len() > 0 {
		elem := q.Remove(q.Front()).([]int)
		for _, dir := range dirs {
			r, c := elem[0]+dir[0], elem[1]+dir[1]
			if r < 0 || r >= len(grid) || c < 0 || c >= len(grid) || grid[r][c] == -1 {
				continue
			}
			if grid[r][c] == 0 {
				borders.PushBack([]int{r, c})
			} else {
				q.PushBack([]int{r, c})
			}
			grid[r][c] = -1
		}
	}
	// ÂØªÊâæÂ≤õÂ±ø
	for roads := 1; borders.Len() > 0; roads++ {
		for i := borders.Len(); i > 0; i-- {
			elem := borders.Remove(borders.Front()).([]int)
			for _, dir := range dirs {
				r, c := elem[0]+dir[0], elem[1]+dir[1]
				if r < 0 || r >= len(grid) || c < 0 || c >= len(grid) || grid[r][c] == -1 {
					continue
				}
				if grid[r][c] == 1 {
					return roads
				}
				borders.PushBack([]int{r, c})
				grid[r][c] = -1
			}
		}
	}
	return 0
}
