func numDecodings(s string) int {
	dp := make([]int, len(s)+1)
	dp[0] = 1
	for i := 0; i < len(s); i++ {
		exist := false // Ê†áËÆ∞ÊòØÂê¶ÂèØ‰ª•Ëß£Á†Å
		// ÊÉÖÂÜµ‰∏Ä
		if s[i] > '0' {

			exist = true
		}
		// ÊÉÖÂÜµ‰∫å
		if i > 0 && s[i-1] != '0' {
			num := 10*int(s[i-1]-'0') + int(s[i]-'0')
			if num > 0 && num <= 26 {
				dp[i+1] += dp[i-1]
				exist = true
			}
		}
		if !exist {
			break
		}
	}
	return dp[len(s)]
}
