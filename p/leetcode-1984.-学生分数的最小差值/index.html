func minimumDifference(nums []int, k int) int {
	min := math.MaxInt32
	sort.Ints(nums)
	for i := k - 1; i < len(nums); i++ {
		if min > nums[i]-nums[i-k+1] {
			min = nums[i] - nums[i-k+1]
		}
	}
	return min
}
