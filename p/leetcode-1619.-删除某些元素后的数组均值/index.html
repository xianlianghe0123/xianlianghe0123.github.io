func trimMean(arr []int) float64 {
	sort.Ints(arr)
	sum := 0
	for _, num := range arr[len(arr)/20 : len(arr)-len(arr)/20] {
		sum += num
	}
	return float64(sum) / float64(len(arr) - len(arr)/10)
}
