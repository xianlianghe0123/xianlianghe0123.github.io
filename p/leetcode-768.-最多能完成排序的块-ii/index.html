func maxChunksToSorted(arr []int) int {
	sorted := append([]int{}, arr...)
	sort.Ints(sorted)
	cnt := make(map[int]int)
	blocks := 0
	for i := range arr {
		cnt[arr[i]]--
		if cnt[arr[i]] == 0 {
			delete(cnt, arr[i])
		}
		cnt[sorted[i]]++
		if cnt[sorted[i]] == 0 {
			delete(cnt, sorted[i])
		}
		if len(cnt) == 0 {
			blocks++
		}
	}
	return blocks
}
