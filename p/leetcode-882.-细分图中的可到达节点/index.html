func reachableNodes(edges [][]int, maxMoves int, n int) int {
	next := make([][][]int, n)
	for _, e := range edges {
		next[e[0]] = append(next[e[0]], e[1:])
		next[e[1]] = append(next[e[1]], []int{e[0], e[2]})
	}

	ans := 0
	minPath := make(map[int]int) // æœ€çŸ­è·¯å¾„
	visit := make([]bool, n)     // å·²æ±‚å‡ºæœ€çŸ­è·¯å¾„çš„èŠ‚ç‚¹é›†åˆ

	h := &MinHeap{
		elems: make([][]int, 0, n),
	}
	heap.Push(h, []int{0, 0})
	for h.Len() > 0 {
		e := heap.Pop(h).([]int)
		u, path := e[0], e[1]
		if visit[u] {
			continue
		}
		visit[u] = true   // æ ‡è®°å·²ç»æ±‚å‡ºæœ€çŸ­è·¯å¾„
		ans++             // å¯åˆ°è¾¾èŠ‚ç‚¹
		minPath[u] = path // ç»´æŠ¤æœ€çŸ­è·¯å¾„
		for _, nextNode := range next[u] {
			v, cnt := nextNode[0], nextNode[1]
			if visit[v] {
				continue
			}
			// å¯è¾¾çš„ä¸‹ä¸€ä¸ªèŠ‚ç‚¹å…¥é˜Ÿ
			if path+cnt+1 <= maxMoves {
				heap.Push(h, []int{v, path + cnt + 1})
			}
		}
	}
	for _, e := range edges {
		steps := 0
		// å¯ä» e[0] å‰è¿›çš„æ­¥æ•°
		if _, ok := minPath[e[0]]; ok {
			steps += maxMoves - minPath[e[0]]
		}
		// å¯ä» e[1] å‰è¿›çš„æ­¥æ•°
		if _, ok := minPath[e[1]]; ok {
			steps += maxMoves - minPath[e[1]]
		}
		// ç»†åˆ†èŠ‚ç‚¹ä¸Šé™
		if steps > e[2] {
			steps = e[2]
		}
		ans += steps
	}
	return ans
}

type MinHeap struct {
	elems [][]int
}

func (m *MinHeap) Len() int {
	return len(m.elems)
}

func (m *MinHeap) Less(i, j int) bool {
	return m.elems[i][1] < m.elems[j][1]
}

func (m *MinHeap) Swap(i, j int) {
	m.elems[i], m.elems[j] = m.elems[j], m.elems[i]
}

func (m *MinHeap) Push(elem interface{}) {
	m.elems = append(m.elems, elem.([]int))
}

func (m *MinHeap) Pop() interface{} {
	tail := m.elems[len(m.elems)-1]
	m.elems = m.elems[:len(m.elems)-1]
	return tail
}
