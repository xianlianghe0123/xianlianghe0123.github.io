func firstBadVersion(n int) int {
	var binSearch func(i, j int) int
	binSearch = func(i, j int) int {
		if i == j {
			return i
		}
		mid := (i + j) / 2
		if isBadVersion(mid) {
			return binSearch(i, mid)
		} else {
			return binSearch(mid+1, j)
		}
	}
	return binSearch(1, n)
}
