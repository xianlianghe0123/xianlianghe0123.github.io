func reversePrefix(word string, ch byte) string {
	for i := range word {
		// ÊâæÂà∞
		if word[i] == ch {
			t := []byte(word)
			// ÂèçËΩ¨
			for j := 0; j < i; j, i = j+1, i-1 {
				t[j], t[i] = t[i], t[j]
			}
			word = string(t)
			break
		}
	}
	return word
}
