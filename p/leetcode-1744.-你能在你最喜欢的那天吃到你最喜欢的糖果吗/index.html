func canEat(candiesCount []int, queries [][]int) []bool {
	sum := make([]int, len(candiesCount))
	sum[0] = candiesCount[0]
	for i := 1; i < len(sum); i++ {
		sum[i] = sum[i-1] + candiesCount[i]
	}
	canEat := make([]bool, 0, len(queries))
	for _, query := range queries {
		EatMin := query[1] + 1
		EatMax := (query[1] + 1) * query[2]
		// Â∞è‰∫éÂ§ßÁöÑÔºåÂ§ß‰∏éÂ∞èÁöÑÊúâ‰∫§ÈõÜ
		if sum[query[0]] >= EatMin && ((query[0] > 0 && sum[query[0]-1] < EatMax) || query[0] == 0) {
			canEat = append(canEat, true)
		} else {
			canEat = append(canEat, false)
		}
	}
	return canEat
}
