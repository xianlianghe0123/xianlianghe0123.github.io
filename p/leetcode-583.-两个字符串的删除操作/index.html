func minDistance(word1 string, word2 string) int {
	dp := make([][]int, len(word1)+1)
	dp[0] = make([]int, len(word2)+1)
	for i := range word1 {
		dp[i+1] = make([]int, len(word2)+1)
		for j := range word2 {
			if word1[i] == word2[j] {
				dp[i+1][j+1] = dp[i][j] + 1
				continue
			}
			dp[i+1][j+1] = dp[i+1][j]
			if dp[i+1][j+1] < dp[i][j+1] {
				dp[i+1][j+1] = dp[i][j+1]
			}
		}
	}
	return len(word1) + len(word2) - (dp[len(word1)][len(word2)] << 1)
}

func minDistance(word1 string, word2 string) int {
	dp := make([][]int, len(word1)+1)
	dp[0] = make([]int, len(word2)+1)
	for j := range word2 {
		dp[0][j+1] = j+1
	}
	for i := range word1 {
		dp[i+1] = make([]int, len(word2)+1)
		dp[i+1][0] = i+1
		for j := range word2 {
			if word1[i] == word2[j] {
				dp[i+1][j+1] = dp[i][j]
				continue
			}
			dp[i+1][j+1] = dp[i+1][j]
			if dp[i+1][j+1] > dp[i][j+1] {
				dp[i+1][j+1] = dp[i][j+1]
			}
			dp[i+1][j+1]++
		}
	}
	return dp[len(word1)][len(word2)]
}
