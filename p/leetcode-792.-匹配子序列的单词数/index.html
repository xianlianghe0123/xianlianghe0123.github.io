func numMatchingSubseq(s string, words []string) int {
	idx := make([][26]int, len(s)+1)
	for i := len(s) - 1; i >= 0; i-- {
		for j := range idx[i+1] {
			if idx[i+1][j] == 0 {
				continue
			}
			idx[i][j] = idx[i+1][j]
		}
		idx[i][s[i]-'a'] = i
	}
	ans := 0
	for _, word := range words {
		j := 0
		for i := range word {
			find := idx[j][word[i]-'a']
			// Êú™ÊâæÂà∞
			if find == 0 && j > 0 {
				break
			}
			j = find + 1
			if i == len(word)-1 {
				ans++
			}
		}
	}
	return ans
}
