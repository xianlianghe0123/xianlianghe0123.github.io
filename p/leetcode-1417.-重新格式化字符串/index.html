func reformat(s string) string {
	digits := make([]byte, 0, len(s))
	letters := make([]byte, 0, len(s))
	for i := range s {
		if s[i] >= '0' && s[i] <= '9' {
			digits = append(digits, s[i])
		} else {
			letters = append(letters, s[i])
		}
	}
	curDigit := true
	switch len(digits) - len(letters) {
	case 0, 1:
	case -1:
		curDigit = false
	default:
		return ""
	}
	ans := make([]byte, 0, len(s))
	for i, j := 0, 0; len(ans) != len(s); curDigit = !curDigit {
		if curDigit {
			ans = append(ans, digits[i])
			i++
		} else {
			ans = append(ans, letters[j])
			j++
		}
	}
	return string(ans)
}
