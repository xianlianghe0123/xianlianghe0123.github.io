func findTargetSumWays(nums []int, target int) int {
	dp := [20][2002]int{}
	dp[0][1000+nums[0]] += 1
	dp[0][1000-nums[0]] += 1
	for i := 1; i < len(nums); i++ {
		for j, cnt := range dp[i-1] {
			if cnt == 0 {
				continue
			}
			dp[i][j+nums[i]] += dp[i-1][j]
			dp[i][j-nums[i]] += dp[i-1][j]
		}
	}
	return dp[len(nums)-1][target+1000]
}
