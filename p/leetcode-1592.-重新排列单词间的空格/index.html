func reorderSpaces(text string) string {
	spaces := 0
	words := 0
	for i := range text {
        // Á©∫Ê†ºÊï∞Èáè
		if text[i] == ' ' {
			spaces++
			continue
		}

		if i == 0 || text[i-1] == ' ' {
			words++
		}
	}
    // Èó¥Èöî
	interval := spaces
	if words > 1 {
		interval = spaces / (words - 1)
	}
	ans := make([]byte, 0, len(text))
	for text != "" {
		text = strings.TrimLeft(text, " ")
        // Ê∑ªÂä†ÂçïËØç
		for ; len(text) > 0 && text[0] != ' '; text = text[1:] {
			ans = append(ans, text[0])
		}
        // Ë°•ÂÖÖÁ©∫Ê†º
		for j := 0; len(ans) < cap(ans) && j < interval; j++ {
			ans = append(ans, ' ')
		}
	}
    // ÊúâÂ§ö‰ΩôÁ©∫Ê†ºÔºåÊ∑ªÂä†Âà∞Êú´Â∞æ
	for len(ans) < cap(ans) {
		ans = append(ans, ' ')
	}
	return string(ans)
}
