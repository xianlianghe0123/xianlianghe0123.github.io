func copyRandomList(head *Node) *Node {
	mapping := make(map[*Node]*Node)
	newHead := new(Node)
	node := newHead
	for ; head != nil; head = head.Next {
		// ‰ºòÂÖàÊü•ÊâæÔºåÂèØËÉΩÂ∑≤Ë¢´randomÊûÑÈÄ†
		newNode, ok := mapping[head]
		if !ok {
			newNode = &Node{
				Val:    head.Val,
				Next:   nil,
				Random: nil,
			}
			mapping[head] = newNode
		}
		node.Next = newNode
		node = node.Next
		// random ÊåáÈíàÊõ¥Êñ∞

			if !ok {

					Next:   nil,
					Random: nil,
				}
				mapping[head.Random] = randomNode
			}
			node.Random = randomNode
		}
	}
	return newHead.Next
}

func copyRandomList(head *Node) *Node {
	if head == nil {
		return nil
	}
	// Ë£ÇÂèò ÁªìÁÇπ
	for node := head; node != nil; node = node.Next.Next {
		node.Next = &Node{
			Val:    node.Val,
			Next:   node.Next,
			Random: node.Random,
		}
	}
	// ‰øÆÊ≠£ random
	for node := head; node != nil; node = node.Next.Next {
		if node.Random != nil {
			node.Next.Random = node.Next.Random.Next
		}
	}
	// ÊãÜÂàÜÈìæË°®
	newHead := head.Next
	for node, newNode := head, newHead; node != nil; node, newNode = node.Next, newNode.Next {
		node.Next = node.Next.Next
		if node.Next != nil {
			newNode.Next = newNode.Next.Next
		}
	}
	return newHead
}
