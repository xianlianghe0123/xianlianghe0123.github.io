func copyRandomList(head *Node) *Node {
	if head == nil {
		return nil
	}
	// è£‚å˜ ç»“ç‚¹

			Val:    node.Val,
			Next:   node.Next,
			Random: node.Random,
		}

		}

		if node.Next != nil {
			newNode.Next = newNode.Next.Next
		}
	}
	return newHead
}
