func isEvenOddTree(root *TreeNode) bool {
	if root == nil {
		return true
	}
	deque := list.New()
	deque.PushBack(root)
	for even := true; deque.Len() > 0; even = !even {

		// ÈÅçÂéÜÊï¥Â±Ç

			if node.Left != nil {

				deque.PushBack(node.Right)
			}
			// ÂÖÉÁ¥†Â•áÂÅ∂ÊÄß‰∏çÊ≠£Á°Æ Êàñ ÂÅ∂Êï∞Â±ÇÈùû‰∏•Ê†ºÈÄíÂ¢û Êàñ Â•áÊï∞Â±ÇÈùû‰∏•Ê†ºÈÄíÂáè

	}
	return true
}
