func canCross(stones []int) bool {
	dp := make([]map[int]bool, len(stones))
	dp[0] = map[int]bool{0: true}
	for i := 0; i < len(stones); i++ {
		for step := range dp[i] {
			for j := i + 1; j < len(stones) && stones[j]-stones[i] <= step+1; j++ {
				if stones[j]-stones[i] >= step-1 {
					if dp[j] == nil {
						dp[j] = make(map[int]bool)
					}
					dp[j][stones[j]-stones[i]] = true
				}
			}
		}
	}
	if len(dp[len(stones)-1]) != 0 {
		return true
	}
	return false
}
