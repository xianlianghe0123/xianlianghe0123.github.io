func countValidWords(sentence string) int {
	cnt := 0
	start := 0
	for i := 0; i <= len(sentence); i++ {
		if i == len(sentence) || sentence[i] == ' ' {
			// Â≠òÂú®ÊúâÊïà token
			if start < i {
				hasHyphen := false
				for j := start; j < i; j++ {
					// Â≠òÂú®Êï∞Â≠óÔºå‰∏çÊòØÊúâÊïàÂçïËØç
					if sentence[j] >= '0' && sentence[j] <= '9' {
						cnt--
						break
					}
					// Ê†áÁÇπÁ¨¶Âè∑‰∏çÂú®Êú´Â∞æÔºå‰∏çÊòØÊúâÊïàÂçïËØç
					if (sentence[j] == ',' || sentence[j] == '.' || sentence[j] == '!') && j != i-1 {
						cnt--
						break
					}
					//
					if sentence[j] == '-' {
						// ËøûÂ≠óÁ¨¶Âú®Â§¥Â∞æ Êàñ ÂêéÈù¢ÊòØÊ†áÁÇπ Êàñ Êï∞ÈáèË∂ÖËøá1
						if j == start || j == i-1 || sentence[j+1] == ',' || sentence[j+1] == '.' || sentence[j+1] == '!' || hasHyphen {
							cnt--
							break
						}
						hasHyphen = true
					}
				}
				cnt++
			}
			start = i + 1
		}
	}
	return cnt
}
