func buildArray(target []int, n int) []string {
	ans := make([]string, 0, n<<1)
	for i := 1; i <= n; i++ {
		ans = append(ans, "Push")
		if target[0] != i {
			ans = append(ans, "Pop")
			continue
		}
		target = target[1:]
		if len(target) == 0 {
			break
		}
	}
	return ans
}
