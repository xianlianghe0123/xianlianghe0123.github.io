func oneEditAway(first string, second string) bool {
	if len(first) > len(second) {
		first, second = second, first
	}
	if len(second)-len(first) > 1 {
		return false
	}
	// Á≠âÈïøÔºåÁªüËÆ°‰∏çÂêåÂ≠óÁ¨¶Êï∞ÈáèÔºåË¶ÅÊ±Ç ‚â§ 1
	if len(first) == len(second) {
		diff := 0
		for i := range first {
			if first[i] != second[i] {
				diff++
				if diff > 1 {
					return false
				}
			}
		}
		return true
	}
	// ‰∏çÁ≠âÈïøÔºåÂØªÊâæÈ¶ñ‰∏™‰∏çÂêåÁöÑÂ≠óÁ¨¶
	i := 0
	for ; i < len(first); i++ {
		if first[i] != second[i] {
			break
		}
	}
	// Âà§Êñ≠ÂêéÁºÄÊòØÂê¶Áõ∏Âêå
	return first[i:] == second[i+1:]
}
