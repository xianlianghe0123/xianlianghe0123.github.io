type node struct {
	val  int
	next []*node
}

type Skiplist struct {
	head        *node
	randomLevel func() int
	level       int
}

func Constructor() Skiplist {
	const maxLevel = 16
	return Skiplist{
		head: &node{
			val:  -1,
			next: make([]*node, maxLevel),
		},
		randomLevel: func() func() int {
			random := rand.New(rand.NewSource(time.Now().UnixNano())).Int31n
			return func() int {
				return maxLevel - int(math.Log2(float64(random(math.MaxInt16))))
			}
		}(),
		level: maxLevel,
	}
}

func (this *Skiplist) Search(target int) bool {

	n := this.head
	for level := this.level - 1; level >= 0; level-- {
		// ÂΩìÂâç level ‰∏çÂ§ß‰∫é target
		for ; n.next[level] != nil && n.next[level].val <= target; n = n.next[level] {
		}
		// Â∑≤ÊâæÂà∞
		if n.val == target {
			return true
		}
	}
	return false
}

func (this *Skiplist) Add(num int) {
	preNodes := make([]*node, this.level) // ÊØèÂ±ÇÂâçÁΩÆËäÇÁÇπ
	for level, n := this.level-1, this.head; level >= 0; level-- {
		// ÂΩìÂâç level ‰∏çÂ§ß‰∫é num
		for ; n.next[level] != nil && n.next[level].val <= num; n = n.next[level] {
		}
		preNodes[level] = n
	}
	level := this.randomLevel() // ÈöèÊú∫ÂΩìÂâçËäÇÁÇπÁöÑ
	newNode := &node{
		val:  num,
		next: make([]*node, level),
	}
	// Êõ¥Êñ∞ÂâçÁΩÆÁªìÁÇπ
	for i := level - 1; i >= 0; i-- {
		preNodes[i].next[i], newNode.next[i] = newNode, preNodes[i].next[i]
	}
}

func (this *Skiplist) Erase(num int) bool {
	n := this.head
	preNodes := make([]*node, this.level) // ÊØèÂ±ÇÂâçÁΩÆËäÇÁÇπ
	for level := this.level - 1; level >= 0; level-- {
		// ÂΩìÂâç level Â∞è‰∫é num
		for ; n.next[level] != nil && n.next[level].val < num; n = n.next[level] {
		}
		preNodes[level] = n
	}
	n = n.next[0]
	// Âà§ÂÆöÊòØÂê¶Â≠òÂú®
	if n == nil || n.val != num {
		return false
	}
	for level := len(n.next) - 1; level >= 0; level-- {
		preNodes[level].next[level] = n.next[level]
	}
	return true
}
