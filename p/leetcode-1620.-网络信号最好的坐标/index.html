func bestCoordinate(towers [][]int, radius int) []int {
	ans := []int{0, 0}
	maxSign := -1
	const Max = 51
	for i := 0; i < Max; i++ {
		for j := 0; j < Max; j++ {
			sign := 0
			for _, tower := range towers {
				d2 := (i-tower[0])*(i-tower[0]) + (j-tower[1])*(j-tower[1])
				if d2 > radius*radius {
					continue
				}
				sign += int(float64(tower[2]) / (1 + math.Sqrt(float64(d2))))
			}
			if maxSign < sign {
				maxSign = sign
				ans = []int{i, j}
			}
		}
	}
	return ans
}
