func findSecondMinimumValue(root *TreeNode) int {
	max := -1
	// éå†
	var dfs func(node *TreeNode)
	dfs = func(node *TreeNode) {
		if node == nil {
			return
		}
		if (max == -1 || max > node.Val) && node.Val != root.Val {
			max = node.Val
		}
		dfs(node.Left)
		dfs(node.Right)
	}
	dfs(root)
	return max
}
