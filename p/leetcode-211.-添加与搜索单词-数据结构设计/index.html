type Node struct {
	end  bool
	char map[byte]*Node
}

type WordDictionary struct {
	head *Node
}

func Constructor() WordDictionary {
	return WordDictionary{
		head: &Node{
			end:  false,
			char: make(map[byte]*Node),
		},
	}
}

func (this *WordDictionary) AddWord(word string) {
	node := this.head
	for i := range word {
		if _, ok := node.char[word[i]]; !ok {
			node.char[word[i]] = &Node{
				end:  false,
				char: make(map[byte]*Node),
			}
		}
		node = node.char[word[i]]
	}
	node.end = true
}

func (this *WordDictionary) Search(word string) bool {
	var search func(node *Node, suffix []byte) bool
	search = func(node *Node, suffix []byte) bool {
		for i, c := range suffix {
			if c == '.' {
                // å°è¯•é€’å½’åŒ¹é…æ‰€æœ‰é¡¹

	}
	return search(this.head, []byte(word))
}
