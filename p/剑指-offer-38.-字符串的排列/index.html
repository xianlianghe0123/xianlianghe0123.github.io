func permutation(s string) []string {
	only := make(map[string]bool)

	choice := make([]bool, len(s))
	prefix := make([]byte, 0, len(s))
	var backtrack func()
	backtrack = func() {
		if len(prefix) == len(s) {
			only[string(prefix)] = true
			return
		}
		for i := range choice {
			if choice[i] {
				continue
			}
			prefix = append(prefix, s[i])
			choice[i] = true
			backtrack()
			choice[i] = false
			prefix = prefix[:len(prefix)-1]
		}
	}
	backtrack()
	res := make([]string, 0, 1<<(len(s)-1))
	for k := range only {
		res = append(res, k)
	}
	return res
}
