func countStudents(students []int, sandwiches []int) int {
	square := 0
	for _, s := range students {
		square += s
	}
	circle := len(students) - square
	for _, sandwich := range sandwiches {
		if sandwich == 0 && circle > 0 {
			circle--
			continue
		}
		if sandwich == 1 && square > 0 {
			square--
			continue
		}
		break
	}
	return square + circle
}
