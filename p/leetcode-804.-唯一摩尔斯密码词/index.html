func uniqueMorseRepresentations(words []string) int {
	exist := make(map[string]struct{})
	for _, word := range words {
		t := make([]byte, 0, len(word)*4)
		for i := range word {
			t = append(t, morse[word[i]-'a']...)
		}
		exist[string(t)] = struct{}{}
	}
	return len(exist)
}
