func countPairs(deliciousness []int) int {
	const MOD = 1e9 + 7
	sum := 0
	foodCnt := make(map[int]int)
	for _, d := range deliciousness {
		foodCnt[d]++
	}
	for d, cnt := range foodCnt {
		for power := 1; power < math.MaxInt32 && power > 0; power <<= 1 {
			another := power - d
			if another < d || foodCnt[another] == 0 {
				continue
			}
			if another == d {
				sum += cnt * (cnt - 1) / 2 % MOD
			} else {
				sum += (foodCnt[another] * cnt) % MOD
			}
			sum %= MOD
		}
	}
	return sum
}
