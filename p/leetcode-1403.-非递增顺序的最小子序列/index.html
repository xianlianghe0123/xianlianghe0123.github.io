func minSubsequence(nums []int) []int {
	halfOfSum := 0
	for _, num := range nums {
		halfOfSum += num
	}
	halfOfSum /= 2
	sort.Sort(sort.Reverse(sort.IntSlice(nums)))
	s := 0
	i := 0
	for i = range nums {
		s += nums[i]
		if s > halfOfSum {
			break
		}
	}
	return nums[:i+1]
}
