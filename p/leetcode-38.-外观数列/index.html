func countAndSay(n int) string {
    item := []byte{'1'}
    for i := 2; i <= n; i++ {
        t := make([]byte, 0, len(item)<<1)
        char := item[0]
        cnt := 1
        for j := 1; j < len(item); j++ {
            if item[j] != char {
                t = append(t, strconv.FormatInt(int64(cnt), 10)...)
                t = append(t, char)
                char = item[j]
                cnt = 1
                continue
            }
            cnt++
        }
        t = append(t, strconv.FormatInt(int64(cnt), 10)...)
        t = append(t, char)
        item = t
    }
    return string(item)
}
